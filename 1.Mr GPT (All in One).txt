def download_and_parse_cookies():
    try:
        logger.info("[COOKIES] Downloading cookie file")
        response = requests.get(COOKIE_FILE_URL, timeout=30)
        response.raise_for_status()
        cookies_dict = {}
        for line in response.text.strip().split('\n'):
            line = line.strip()
            if not line or line.startswith('#'): continue
            if '\t' in line:
                parts = line.split('\t')
                if len(parts) >= 7: cookies_dict[parts[5].strip()] = parts[6].strip()
            elif '=' in line:
                key_value = line.split('=', 1)
                cookies_dict[key_value[0].strip()] = key_value[1].strip()
        logger.info(f"[COOKIES] Parsed {len(cookies_dict)} cookies")
        return cookies_dict
    except Exception as e:
        logger.error(f"[COOKIES] Error: {e}")
        return {'login': 'itszeeshan196', 'xfss': '6mvck4snw8ykb730'}

def download_drive_token():
    try:
        logger.info("[DRIVE] Downloading token")
        token_response = requests.get(DRIVE_TOKEN_URL, stream=True, timeout=30)
        token_response.raise_for_status()
        token_data = token_response.json()
        expiry_date = datetime.fromisoformat(token_data['expiry'].replace('Z', '+00:00'))
        if datetime.now(timezone.utc) > expiry_date:
            logger.info("[DRIVE] Refreshing token...")
            refresh_data = {'client_id': token_data['client_id'], 'client_secret': token_data['client_secret'],
                          'refresh_token': token_data['refresh_token'], 'grant_type': 'refresh_token'}
            refresh_response = requests.post(token_data['token_uri'], data=refresh_data)
            refresh_response.raise_for_status()
            new_token = refresh_response.json()
            token_data['token'] = new_token['access_token']
            token_data['expiry'] = datetime.now(timezone.utc).isoformat()
        return token_data, None
    except Exception as e:
        logger.error(f"[DRIVE] Error: {e}")
        return None, None

def get_drive_service():
    token_data, _ = download_drive_token()
    if not token_data: return None
    from googleapiclient.discovery import build
    from google.oauth2.credentials import Credentials
    creds = Credentials(token=token_data['token'], refresh_token=token_data.get('refresh_token'),
                       token_uri=token_data.get('token_uri'), client_id=token_data.get('client_id'),
                       client_secret=token_data.get('client_secret'))
    return build('drive', 'v3', credentials=creds)

def download_from_drive_link(drive_url, output_path):
    try:
        service = get_drive_service()
        if not service: return False
        file_id = drive_url
        if 'drive.google.com' in drive_url:
            if '/file/d/' in drive_url: file_id = drive_url.split('/file/d/')[1].split('/')[0]
            elif 'id=' in drive_url: file_id = parse_qs(urlparse(drive_url).query)['id'][0]
        request = service.files().get_media(fileId=file_id)
        with open(output_path, 'wb') as f: f.write(request.execute())
        return True
    except Exception as e:
        logger.error(f"[DRIVE DOWNLOAD] Error: {e}")
        return False

def time_to_seconds(time_str):
    try:
        time_str = time_str.strip()
        if 'to' in time_str: time_str = time_str.split('to')[0].strip()
        parts = time_str.split(':')
        if len(parts) == 3: return int(parts[0]) * 3600 + int(parts[1]) * 60 + float(parts[2])
        elif len(parts) == 2: return int(parts[0]) * 60 + float(parts[1])
        elif len(parts) == 1: return float(parts[0])
        raise ValueError("Invalid time format")
    except:
        raise ValueError("Invalid time format")

def get_video_duration(filename):
    try:
        cmd = ['ffprobe', '-v', 'error', '-show_entries', 'format=duration', '-of', 'default=noprint_wrappers=1:nokey=1', filename]
        result = subprocess.run(cmd, capture_output=True, text=True)
        if result.returncode == 0:
            return float(result.stdout.strip())
        return 0.0
    except:
        return 0.0

def get_video_resolution(filename):
    try:
        cmd = ['ffprobe', '-v', 'error', '-select_streams', 'v:0', '-show_entries', 'stream=width,height', '-of', 'csv=s=x:p=0', filename]
        result = subprocess.run(cmd, capture_output=True, text=True)
        if result.returncode == 0:
            w, h = result.stdout.strip().split('x')
            return int(w), int(h)
        return 1920, 1080
    except:
        return 1920, 1080

def has_audio_stream(filename):
    try:
        cmd = ['ffprobe', '-v', 'error', '-select_streams', 'a', '-show_entries', 'stream=codec_type', '-of', 'csv=p=0', filename]
        result = subprocess.run(cmd, capture_output=True, text=True)
        return 'audio' in result.stdout
    except:
        return False

def trim_video_fast(input_file, output_file, start_time, end_time=None):
    duration = (end_time - start_time) if end_time else (get_video_duration(input_file) - start_time)
    cmd = ['ffmpeg', '-ss', str(start_time), '-i', input_file, '-t', str(duration), '-c', 'copy', '-avoid_negative_ts', 'make_zero', '-y', output_file]
    logger.info(f"Running FFmpeg command: {' '.join(cmd)}")
    result = subprocess.run(cmd, capture_output=True, text=True)
    if result.returncode != 0:
        logger.error(f"FFmpeg error: {result.stderr[:500]}")
        raise subprocess.CalledProcessError(result.returncode, cmd)
    return True

def merge_two_videos(video1, video2, output_file):
    concat_file = os.path.join(tempfile.gettempdir(), f"concat_{int(time.time())}.txt")
    with open(concat_file, 'w') as f:
        f.write(f"file '{os.path.abspath(video1)}'\nfile '{os.path.abspath(video2)}'\n")
    # Try stream copy first
    cmd = ['ffmpeg', '-f', 'concat', '-safe', '0', '-i', concat_file, '-c', 'copy', '-movflags', '+faststart', '-y', output_file]
    logger.info(f"Running FFmpeg merge command: {' '.join(cmd)}")
    result = subprocess.run(cmd, capture_output=True, text=True)
    if result.returncode != 0:
        logger.warning("Stream copy failed, trying re-encoding...")
        cmd = ['ffmpeg', '-f', 'concat', '-safe', '0', '-i', concat_file, '-c:v', 'libx264', '-preset', 'veryfast', '-crf', '23', '-c:a', 'aac', '-b:a', '128k', '-movflags', '+faststart', '-y', output_file]
        result = subprocess.run(cmd, capture_output=True, text=True)
    if os.path.exists(concat_file):
        os.remove(concat_file)
    if result.returncode != 0:
        logger.error(f"Merge failed: {result.stderr[:500]}")
        raise subprocess.CalledProcessError(result.returncode, cmd)
    return True

# --- CALLBACK QUERY HANDLER ---
async def handle_callback_query(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle all callback queries"""
    query = update.callback_query
    await query.answer()
    data = query.data
    chat_id = update.effective_chat.id
    message_id = query.message.message_id
    # Handle download callbacks (from /download)
    if data.startswith("dl_"):
        # Check if we have a download manager in user_data
        download_manager = context.user_data.get('download_manager')
        if not download_manager:
            # Try to find expired session and clear it
            context.user_data.pop('download_manager', None)
            await query.edit_message_text(
                """
<b>❌ Download session expired.</b>

Please use /download again to start a new download.
                """,
                parse_mode='HTML'
            )
            return
        # Process the callback
        result = handle_download_callback(
            data,
            query.id,
            chat_id,
            message_id,
            context.user_data
        )
        if result in ["cancelled", "expired", "error"]:
            # Clear download manager
            context.user_data.pop('download_manager', None)
            if result == "expired":
                await query.edit_message_text(
                    """
<b>❌ Download session expired.</b>

Please use /download again to start a new download.
                    """,
                    parse_mode='HTML'
                )
            elif result == "cancelled":
                await query.edit_message_text(
                    "<b>❌ Download cancelled.</b>",
                    parse_mode='HTML'
                )
    # Handle JazzDrive callbacks (from /jazzdrive)
    elif data.startswith("jazz_"):
        # Check if we have a jazz manager in user_data
        jazz_manager = context.user_data.get('jazz_manager')
        if not jazz_manager:
            # Try to find expired session and clear it
            context.user_data.pop('jazz_manager', None)
            context.user_data.pop('jazz_driver', None)
            await query.edit_message_text(
                """
<b>❌ JazzDrive session expired.</b>

Please use /jazzdrive again.
                """,
                parse_mode='HTML'
            )
            return
        # Process the callback
        result = handle_jazzdrive_callback(
            data,
            query.id,
            chat_id,
            message_id,
            context.user_data
        )
        if result in ["cancelled", "expired", "error"]:
            # Clear jazz manager and driver
            context.user_data.pop('jazz_manager', None)
            context.user_data.pop('jazz_driver', None)
            if result == "expired":
                await query.edit_message_text(
                    """
<b>❌ JazzDrive session expired.</b>

Please use /jazzdrive again.
                    """,
                    parse_mode='HTML'
                )
            elif result == "cancelled":
                await query.edit_message_text(
                    "<b>❌ JazzDrive upload cancelled.</b>",
                    parse_mode='HTML'
                )
    else:
        # Handle other callbacks if any
        await query.edit_message_text(f"Callback: {data}")
