# Video processing functions
def run_ffmpeg_command(cmd):
    """Run ffmpeg command and return success status"""
    try:
        subprocess.run(cmd, check=True, capture_output=True, text=True)
        return True, None
    except subprocess.CalledProcessError as e:
        return False, e.stderr

def merge_videos(input_files, output_path):
    """Merge multiple videos"""
    temp_list = os.path.join(DOWNLOADS_DIR, f"inputs_{uuid.uuid4().hex}.txt")
    with open(temp_list, "w") as f:
        for fpath in input_files:
            f.write(f"file '{fpath}'\n")
    success, error = run_ffmpeg_command(
        ["ffmpeg", "-y", "-f", "concat", "-safe", "0", "-i", temp_list, "-c", "copy", output_path]
    )
    os.remove(temp_list)
    return success, error

def trim_video(input_path, output_path, start_time, end_time):
    """Trim video"""
    return run_ffmpeg_command(
        ["ffmpeg", "-y", "-i", input_path, "-ss", start_time, "-to", end_time, "-c", "copy", output_path]
    )

def compress_video(input_path, output_path, quality="medium"):
    """Compress video"""
    crf = {"low": 28, "medium": 23, "high": 18}.get(quality, 23)
    return run_ffmpeg_command(
        ["ffmpeg", "-y", "-i", input_path, "-vcodec", "libx264", "-crf", str(crf), output_path]
    )

def remove_audio(input_path, output_path):
    """Remove audio from video"""
    return run_ffmpeg_command(
        ["ffmpeg", "-y", "-i", input_path, "-an", "-c:v", "copy", output_path]
    )

def add_audio_with_loop(input_path, audio_path, output_path):
    """
    Add audio to video with looping if audio is shorter than video
    Uses complex filter to loop audio to match video duration
    """
    # First get video duration
    cmd = ["ffprobe", "-v", "error", "-show_entries", "format=duration", "-of", "default=noprint_wrappers=1:nokey=1", input_path]
    result = subprocess.run(cmd, capture_output=True, text=True)
    video_duration = float(result.stdout.strip())
    
    # Create filter complex: loop audio to match video duration
    # amovie=audio.mp3:loop=0,asetpts=N/SR/TB[audio] means loop audio infinitely and adjust timestamps
    filter_complex = f"amovie={audio_path}:loop=0,asetpts=N/SR/TB[a]"
    
    return run_ffmpeg_command(
        ["ffmpeg", "-y", "-i", input_path, "-filter_complex", filter_complex, 
         "-map", "0:v", "-map", "[a]", "-c:v", "copy", "-c:a", "aac", 
         "-shortest", "-t", str(video_duration), output_path]
    )

def change_video_speed(input_path, output_path, speed_factor):
    """Change video speed (0.25x to 4x)"""
    # Adjust video speed
    video_filter = f"setpts={1/float(speed_factor)}*PTS"
    # Adjust audio speed
    audio_filter = f"atempo={min(float(speed_factor), 2.0)}"
    
    # If speed > 2.0, need multiple atempo filters (max 2.0 per filter)
    if float(speed_factor) > 2.0:
        # For speeds > 2.0, we need multiple atempo filters
        factors = []
        remaining = float(speed_factor)
        while remaining > 2.0:
            factors.append("atempo=2.0")
            remaining /= 2.0
        factors.append(f"atempo={remaining}")
        audio_filter = ",".join(factors)
    
    filter_complex = f"[0:v]{video_filter}[v];[0:a]{audio_filter}[a]"
    
    return run_ffmpeg_command(
        ["ffmpeg", "-y", "-i", input_path, "-filter_complex", filter_complex,
         "-map", "[v]", "-map", "[a]", "-c:v", "libx264", "-c:a", "aac", output_path]
    )

def crop_video(input_path, output_path, width, height):
    """Crop video to custom dimensions"""
    return run_ffmpeg_command(
        ["ffmpeg", "-y", "-i", input_path, "-vf", f"crop={width}:{height}", "-c:a", "copy", output_path]
    )

def crop_aspect_ratio(input_path, output_path, aspect_ratio):
    """Crop video to specific aspect ratio (e.g., 9:16, 16:9, 1:1)"""
    ratios = {
        "9:16": "ih*9/16:ih",
        "16:9": "iw:iw*9/16",
        "1:1": "min(iw,ih):min(iw,ih)",
        "4:3": "iw:iw*3/4",
        "3:4": "ih*3/4:ih"
    }
    crop_filter = ratios.get(aspect_ratio, "iw:ih")
    return run_ffmpeg_command(
        ["ffmpeg", "-y", "-i", input_path, "-vf", f"crop={crop_filter}", "-c:a", "copy", output_path]
    )

def extract_audio(input_path, output_path):
    """Extract audio from video"""
    return run_ffmpeg_command(
        ["ffmpeg", "-y", "-i", input_path, "-q:a", "0", "-map", "a", output_path]
    )
