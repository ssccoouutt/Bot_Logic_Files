def extract_drive_url(text):
    """Extract Google Drive URL from text - FIXED VERSION"""
    # More comprehensive patterns for Google Drive URLs
    patterns = [
        # Standard file links with /d/ format
        r'(https?://drive\.google\.com/file/d/([a-zA-Z0-9_-]+)/view[^\s]*)',
        r'(https?://drive\.google\.com/file/d/([a-zA-Z0-9_-]+)[^\s]*)',
        # Open links
        r'(https?://drive\.google\.com/open\?id=([a-zA-Z0-9_-]+)[^\s]*)',
        # UC links
        r'(https?://drive\.google\.com/uc\?id=([a-zA-Z0-9_-]+)[^\s]*)',
        # Usercontent links
        r'(https?://drive\.usercontent\.google\.com/download\?id=([a-zA-Z0-9_-]+)[^\s]*)',
        # Any drive.google.com URL
        r'(https?://[^\s]+?drive\.google\.com[^\s]+)'
    ]
    
    debug_print(f"🔍 Attempting to extract URL from: {text[:100]}", "DEBUG")
    
    for i, pattern in enumerate(patterns):
        match = re.search(pattern, text, re.IGNORECASE)
        if match:
            url = match.group(1)
            # Clean up the URL (remove trailing punctuation)
            url = re.sub(r'[.,;:!?)$]$', '', url)
            
            # Extract file ID for verification
            file_id_match = re.search(r'(?:/d/|id=)([a-zA-Z0-9_-]+)', url)
            if file_id_match:
                file_id = file_id_match.group(1)
                debug_print(f"📋 Extracted file ID: {file_id}", "DEBUG")
            
            debug_print(f"✅ Extracted URL (pattern {i+1}): {url}", "SUCCESS")
            return url
    
    debug_print(f"❌ No drive URL found in: {text[:100]}", "DEBUG")
    return None

def is_google_drive_link(text):
    """Check if text contains a Google Drive link"""
    # Pattern for Google Drive links
    patterns = [
        r'drive\.google\.com/file/d/[a-zA-Z0-9_-]+',
        r'drive\.google\.com/open\?id=[a-zA-Z0-9_-]+',
        r'drive\.google\.com/uc\?id=[a-zA-Z0-9_-]+',
        r'drive\.usercontent\.google\.com/download\?id=[a-zA-Z0-9_-]+',
        r'drive\.google\.com'
    ]
    
    for pattern in patterns:
        if re.search(pattern, text, re.IGNORECASE):
            debug_print(f"✅ Drive link detected with pattern: {pattern}", "DEBUG")
            return True
    return False

def get_direct_download_link(drive_url):
    """Get direct download link from Google Drive URL using API - IMPROVED VERSION"""
    try:
        debug_print(f"🔄 Getting direct link for: {drive_url}", "INFO")
        
        # Extract file ID if needed (the API might work better with just the ID)
        file_id_match = re.search(r'(?:/d/|id=)([a-zA-Z0-9_-]+)', drive_url)
        if file_id_match:
            file_id = file_id_match.group(1)
            debug_print(f"📋 Extracted file ID: {file_id}", "DEBUG")
            
            # Try with just the ID first (faster)
            id_url = f"https://drive.google.com/uc?id={file_id}"
            params = {'url': id_url}
            debug_print(f"🔄 Using ID-based URL: {id_url}", "DEBUG")
        else:
            params = {'url': drive_url}
            debug_print(f"🔄 Using original URL: {drive_url}", "DEBUG")
        
        headers = {'ngrok-skip-browser-warning': 'true'}
        
        # API call with timeout (60 seconds as specified)
        debug_print(f"⏳ Calling API with params: {params}", "DEBUG")
        response = requests.get(DRIVE_API_URL, params=params, headers=headers, timeout=60)
        
        debug_print(f"📡 API response status: {response.status_code}", "DEBUG")
        
        if response.status_code == 200:
            try:
                data = response.json()
                debug_print(f"📦 API response data: {data}", "DEBUG")
                
                if data.get('success') and 'direct_link' in data:
                    direct_link = data['direct_link']
                    debug_print(f"✅ Got direct link: {direct_link}", "SUCCESS")
                    return direct_link
                else:
                    debug_print(f"❌ API returned error: {data.get('error', 'Unknown error')}", "ERROR")
            except json.JSONDecodeError as e:
                debug_print(f"❌ Failed to parse API response: {e}", "ERROR")
                debug_print(f"Response text: {response.text[:200]}", "ERROR")
        else:
            debug_print(f"❌ API request failed: {response.status_code}", "ERROR")
            debug_print(f"Response: {response.text[:200]}", "ERROR")
            
    except requests.exceptions.Timeout:
        debug_print(f"⏱️ API timeout after 60 seconds", "WARNING")
    except Exception as e:
        debug_print(f"❌ Error getting direct link: {e}", "ERROR")
    
    return None

def download_and_extract_session():
    """Download session zip and extract it"""
    try:
        debug_print("Loading session...", "INFO")
        response = requests.get(SESSION_URL, stream=True)
        response.raise_for_status()
        
        zip_path = os.path.join(tempfile.gettempdir(), "session.zip")
        with open(zip_path, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                f.write(chunk)
        
        if os.path.exists(CHROME_PROFILE_DIR):
            shutil.rmtree(CHROME_PROFILE_DIR)
        os.makedirs(CHROME_PROFILE_DIR, exist_ok=True)
        
        with zipfile.ZipFile(zip_path, 'r') as zipf:
            for name in zipf.namelist():
                if name.startswith('chrome_profile/'):
                    target_path = os.path.join(CHROME_PROFILE_DIR, name.replace('chrome_profile/', '', 1))
                    os.makedirs(os.path.dirname(target_path), exist_ok=True)
                    if not name.endswith('/'):
                        with zipf.open(name) as source, open(target_path, 'wb') as target:
                            shutil.copyfileobj(source, target)
        
        os.remove(zip_path)
        debug_print("Session loaded", "SUCCESS")
        return True
        
    except Exception as e:
        debug_print(f"Session load failed: {e}", "ERROR")
        return False

def ultra_fast_reply(chat_name, reply_text):
    """Send reply in under 500ms"""
    try:
        start = time.time()
        debug_print(f"⚡ Attempting reply to '{chat_name}'", "DEBUG")
        
        if not chat_name:
            debug_print(f"❌ Cannot reply: chat_name is None", "ERROR")
            return False
        
        # Try multiple methods to click the chat
        try:
            chat_element = driver.find_element(By.XPATH, f'//span[@title="{chat_name}"]')
            chat_element.click()
        except:
            try:
                driver.execute_script(f"""
                    var spans = document.evaluate("//span[@title='{chat_name}']", document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null);
                    if(spans.singleNodeValue) {{
                        spans.singleNodeValue.click();
                    }}
                """)
            except Exception as e:
                debug_print(f"Click failed: {e}", "ERROR")
                return False
        
        time.sleep(0.1)
        
        # Find input box
        input_box = driver.find_element(By.XPATH, '//div[@contenteditable="true"][@data-tab="10"]')
        
        # Clear and send message - use plain text only
        input_box.clear()
        
        # Send the link as plain text without any markdown formatting
        input_box.send_keys(reply_text)
        time.sleep(0.1)
        input_box.send_keys(Keys.ENTER)
        
        elapsed = (time.time() - start) * 1000
        debug_print(f"✅ Replied to '{chat_name}' in {elapsed:.0f}ms", "SUCCESS")
        return True
        
    except Exception as e:
        debug_print(f"Reply failed: {e}", "ERROR")
        return False

async def get_screenshot(update: Update, context: ContextTypes.DEFAULT_TYPE):
    try:
        await context.bot.send_chat_action(chat_id=update.effective_chat.id, action="upload_photo")
        screenshot_bytes = driver.get_screenshot_as_png()
        img = Image.open(io.BytesIO(screenshot_bytes))
        if img.mode != 'RGB':
            img = img.convert('RGB')
        output = io.BytesIO()
        img.save(output, format='JPEG', quality=85)
        
        await update.message.reply_photo(
            photo=output.getvalue(),
            caption=f"📱 WhatsApp - {datetime.now().strftime('%H:%M:%S')}"
        )
    except Exception as e:
        await update.message.reply_text(f"❌ {e}")

        )
        return
    
    drive_url = context.args[0]
    await update.message.reply_text(f"🔄 Testing drive link: {drive_url}\n⏳ This may take up to 60 seconds...")
    
    # First test extraction
    extracted = extract_drive_url(drive_url)
    if not extracted:
        await update.message.reply_text("❌ Failed to extract Google Drive URL from your input")
        return
    
    await update.message.reply_text(f"✅ Extracted URL: {extracted}\n⏳ Getting direct download link...")
    
    # Get direct link
    direct_link = get_direct_download_link(extracted)
    
    if direct_link:
        await update.message.reply_text(
            f"✅ *Direct Download Link:*\n{direct_link}",

        )
    else:
        await update.message.reply_text(
            "❌ Failed to get direct download link.\n"
