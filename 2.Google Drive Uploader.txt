import os
import sys
import json
import subprocess
import importlib.util
from datetime import datetime, timezone
from urllib.parse import unquote

def install_dependencies():
    """Install required packages if not already installed"""
    required_packages = {
        'requests': 'requests',
        'python-telegram-bot': 'python-telegram-bot==20.7'
    }
    
    print("🔍 Checking dependencies...")
    
    for package_name, install_name in required_packages.items():
        try:
            # Try to import the package
            if package_name == 'python-telegram-bot':
                # Check for telegram module specifically
                spec = importlib.util.find_spec('telegram')
                if spec is None:
                    raise ImportError
            else:
                spec = importlib.util.find_spec(package_name)
                if spec is None:
                    raise ImportError
            print(f"✅ {package_name} is already installed")
        except ImportError:
            print(f"📦 Installing {package_name}...")
            try:
                # Try to install the package
                subprocess.check_call([
                    sys.executable, "-m", "pip", "install", 
                    install_name
                ])
                print(f"✅ Successfully installed {package_name}")
            except subprocess.CalledProcessError as e:
                print(f"❌ Failed to install {package_name}: {e}")
                print(f"Please install manually: pip install {install_name}")
                return False
    
    print("✅ All dependencies are installed and ready!")
    return True

# Install dependencies first
if not install_dependencies():
    print("\n❌ Failed to install dependencies. Please install manually:")
    print("pip install requests python-telegram-bot==20.7")
    sys.exit(1)

# Store user states
user_states = {}

def download_and_upload_to_drive(file_url=None):
    """Original download and upload function"""
    # Hardcoded token.json file URL
    token_url = "https://drive.usercontent.google.com/download?id=1NZ3NvyVBnK85S8f5eTZJS5uM5c59xvGM&export=download"
    
    # If file_url is not provided, ask for it
    if file_url is None:
        print("Step 2: Download and upload file to Google Drive")
        print("Please provide the direct download URL of the file you want to upload:")
        file_url = input().strip()
    
    if not file_url:
        return "❌ No file URL provided. Exiting."
    
    try:
        print("Step 1: Downloading token.json from Google Drive...")
        
        # Download token.json from Google Drive
        token_response = requests.get(token_url, stream=True)
        token_response.raise_for_status()
        
        # Extract filename from URL
        token_filename = unquote(os.path.basename(token_url.split('?')[0]))
        
        # If no extension, use default name
        if '.' not in token_filename:
            token_filename = "token.json"
        
        # Save token.json locally
        with open(token_filename, 'wb') as f:
            for chunk in token_response.iter_content(chunk_size=8192):
                if chunk:
                    f.write(chunk)
        
        print(f"Successfully downloaded token file: {token_filename}")
        
        # Load token data
        with open(token_filename, 'r') as f:
            token_data = json.load(f)
        
        print("Token data loaded successfully")
        
        # Check if token is expired
        expiry_date = datetime.fromisoformat(token_data['expiry'].replace('Z', '+00:00'))
        if datetime.now(timezone.utc) > expiry_date:
            print("Token expired. Refreshing...")
            # Refresh the token
            refresh_data = {
                'client_id': token_data['client_id'],
                'client_secret': token_data['client_secret'],
                'refresh_token': token_data['refresh_token'],
                'grant_type': 'refresh_token'
            }
            
            refresh_response = requests.post(token_data['token_uri'], data=refresh_data)
            refresh_response.raise_for_status()
            
            new_token = refresh_response.json()
            token_data['token'] = new_token['access_token']
            token_data['expiry'] = datetime.now(timezone.utc).isoformat()
            
            print("Token refreshed successfully")
        
        print(f"\nDownloading file from: {file_url}")
        
        # Extract filename from URL
        filename = unquote(os.path.basename(file_url.split('?')[0]))
        # If filename has no extension, try to get it from Content-Disposition header
        if '.' not in filename:
            response = requests.head(file_url, allow_redirects=True)
            if 'Content-Disposition' in response.headers:
                content_disposition = response.headers['Content-Disposition']
                if 'filename=' in content_disposition:
                    suggested_name = content_disposition.split('filename=')[1].strip('"\'')
                    if '.' in suggested_name:
                        filename = suggested_name
        
        print(f"File will be saved as: {filename}")
        # Download the file
        file_response = requests.get(file_url, stream=True)
        file_response.raise_for_status()
        # Save to local storage
        total_size = 0
        with open(filename, 'wb') as f:
            for chunk in file_response.iter_content(chunk_size=8192):
                if chunk:
                    f.write(chunk)
                    total_size += len(chunk)
        file_size = total_size / (1024*1024)
        print(f"Successfully downloaded: {filename}")
        print(f"File size: {file_size:.2f} MB")
        # Upload to Google Drive using the token
        print("\nStep 3: Uploading to Google Drive...")
        
        # Upload using Google Drive API
        upload_url = "https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart"
        
        headers = {
            "Authorization": f"Bearer {token_data['token']}"
        }
        
        metadata = {
            "name": filename,
            "parents": ["root"]  # Upload to Drive root
        }
        
        files = {
            "data": ("metadata", json.dumps(metadata), "application/json"),
            "file": (filename, open(filename, "rb"))
        }
        
        upload_response = requests.post(
            upload_url,
            headers=headers,
            files=files
        )
        
        upload_response.raise_for_status()
        
        result = upload_response.json()
        
        # Close the file
        open(filename, "rb").close()
        
        success_message = (
            f"✅ File Successfully Uploaded to Google Drive!\n\n"
            f"📁 File Name: {filename}\n"
            f"📊 File Size: {file_size:.2f} MB\n"
            f"🆔 File ID: {result['id']}\n\n"
            f"🔗 View on Google Drive: https://drive.google.com/file/d/{result['id']}/view\n\n"
            f"📥 Download URL:\n"
            f"https://drive.google.com/uc?export=download&id={result['id']}"
        )
        
        print(f"\n✅ File successfully uploaded to Google Drive!")
        print(f"File name: {filename}")
        print(f"File ID: {result['id']}")
        print(f"View file: https://drive.google.com/file/d/{result['id']}/view")
        
        return success_message
    except requests.exceptions.RequestException as e:
        error_msg = f"❌ Network Error: {str(e)}"
        print(f"Error: {e}")
        return error_msg
    except json.JSONDecodeError:
        error_msg = "❌ Error: Downloaded token file is not valid JSON"
        print(error_msg)
        return error_msg
    except Exception as e:
        error_msg = f"❌ Error occurred: {str(e)}"
        print(f"An error occurred: {e}")
        return error_msg
    finally:
        # Clean up temporary files
        if 'token_filename' in locals() and os.path.exists(token_filename):
            os.remove(token_filename)
        if 'filename' in locals() and os.path.exists(filename):
            try:
                os.remove(filename)
            except:
                pass

# TELEGRAM BOT HANDLERS
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Send a message when the command /start is issued."""
    user_id = update.effective_user.id
    user_states[user_id] = {'state': 'idle'}
    
    await update.message.reply_text(
        "🤖 *Google Drive Uploader Bot*\n\n"
        "I can download files from URLs and upload them to Google Drive!\n\n"
        "📋 *Commands:*\n"
        "/upload - Upload a file from URL to Google Drive\n"
        "/help - Show help information\n"
        "/cancel - Cancel current operation\n\n"
        "Just send /upload and then paste your file URL!",
        parse_mode='Markdown'
    )

async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Send a message when the command /help is issued."""
    help_text = (
        "📚 *Google Drive Uploader Bot - Help*\n\n"
        "*How to use:*\n"
        "1. Send `/upload` command\n"
        "2. Paste the direct download URL\n"
        "3. The bot will download and upload to Google Drive\n\n"
        "*Example URLs:*\n"
        "• `https://example.com/file.zip`\n"
        "• `https://drive.google.com/uc?export=download&id=FILE_ID`\n"
        "• Any direct download link\n\n"
        "*Commands:*\n"
        "/upload - Start upload process\n"
        "/help - Show this help\n"
        "/cancel - Cancel current operation"
    )
    await update.message.reply_text(help_text, parse_mode='Markdown')

async def upload_command(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Handle /upload command"""
    user_id = update.effective_user.id
    user_states[user_id] = {'state': 'waiting_url'}
    
    await update.message.reply_text(
        "📤 *Ready to upload!*\n\n"
        "Please send me the **direct download URL** of the file.\n\n"
        "*Example:* `https://example.com/file.zip`\n\n"
        "I'll download it and upload to Google Drive.\n"
        "Send /cancel to abort.",
        parse_mode='Markdown'
    )

async def cancel_command(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Handle /cancel command"""
    user_id = update.effective_user.id
    if user_id in user_states:
        user_states[user_id] = {'state': 'idle'}
    
    await update.message.reply_text(
        "❌ Operation cancelled.",
        parse_mode='Markdown'
    )

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Handle incoming messages"""
    user_id = update.effective_user.id
    message_text = update.message.text
    
    # Check if user is waiting for URL
    if user_id in user_states and user_states[user_id]['state'] == 'waiting_url':
        if message_text.startswith('http://') or message_text.startswith('https://'):
            user_states[user_id] = {'state': 'processing'}
            
            await update.message.reply_text(
                "✅ URL received!\n\n"
                "Starting download and upload process...\n"
                "Please wait, this may take a while depending on file size.",
                parse_mode='Markdown'
            )
            
            # Run the download/upload process
            try:
                result = await asyncio.to_thread(download_and_upload_to_drive, message_text)
                await update.message.reply_text(
                    result,
                    parse_mode='Markdown',
                    disable_web_page_preview=True
                )
                user_states[user_id] = {'state': 'idle'}
            except Exception as e:
                await update.message.reply_text(
                    f"❌ Error occurred: {str(e)}",
                    parse_mode='Markdown'
                )
                user_states[user_id] = {'state': 'idle'}
        else:
            await update.message.reply_text(
                "❌ Please send a valid URL starting with http:// or https://\n\n"
                "Send /cancel to abort.",
                parse_mode='Markdown'
            )
    else:
        # If not in upload process, show help
        await update.message.reply_text(
            "Send /upload to start uploading a file to Google Drive.\n"
            "Send /help for more information."
        )

def run_bot() -> None:
    """Start the bot."""
    # Create the Application
    application = Application.builder().token(TELEGRAM_BOT_TOKEN).build()
    
    # Register command handlers
    application.add_handler(CommandHandler("start", start))
    application.add_handler(CommandHandler("help", help_command))
    application.add_handler(CommandHandler("upload", upload_command))
    application.add_handler(CommandHandler("cancel", cancel_command))
    
    # Register message handler
    application.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
    
    # Start the Bot
    print("="*60)
    print("🤖 Google Drive Uploader Telegram Bot")
    print("="*60)
    print(f"📱 Bot Token: {TELEGRAM_BOT_TOKEN[:15]}...")
    print("🔄 Starting bot polling...")
    print("📲 Go to Telegram and send /start to your bot")
    print("="*60)
    
    # Run polling
    application.run_polling(allowed_updates=Update.ALL_TYPES, drop_pending_updates=True)

def main():
    """Main function"""
    try:
        run_bot()
    except KeyboardInterrupt:
        print("\n\n🤖 Bot stopped by user.")
        sys.exit(0)
    except Exception as e:
        print(f"❌ Error running bot: {e}")
        sys.exit(1)

if __name__ == '__main__':
    main()
