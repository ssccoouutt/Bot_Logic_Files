def install_dependencies():

                print(f"Please install manually: pip install {install_name}")
                return False
    
    print("✅ All dependencies are installed and ready!")
    return True

# Function to extract video ID from YouTube URL
def extract_video_id(url):
    """Extract YouTube video ID from various URL formats"""
    patterns = [
        r'(?:youtube\.com/watch\?v=|youtu\.be/|youtube\.com/embed/|youtube\.com/v/)([a-zA-Z0-9_-]{11})',
        r'(?:youtube\.com/watch\?.*v=)([a-zA-Z0-9_-]{11})',
    ]
    
    for pattern in patterns:
        match = re.search(pattern, url)
        if match:
            return match.group(1)
    return None

# Function to get video info and download links
def get_youtube_video_info(video_url, quality='best'):
    """
    Get YouTube video information and download links
    
    Args:
        video_url: YouTube video URL
        quality: 'best', 'worst', or specific format ID
    
    Returns:
        Dictionary with video info and download links
    """
    try:
        # YouTube DL options
        ydl_opts = {
            'quiet': True,
            'no_warnings': True,
            'extract_flat': False,
            'format': 'best' if quality == 'best' else 'worst' if quality == 'worst' else quality,
        }
        
        with yt_dlp.YoutubeDL(ydl_opts) as ydl:
            # Extract video info
            info = ydl.extract_info(video_url, download=False)
            
            # Get all available formats
            formats = info.get('formats', [])
            
            # Find the best video+audio format
            best_format = None
            video_formats = []
            
            for f in formats:
                format_info = {
                    'format_id': f.get('format_id'),
                    'ext': f.get('ext'),
                    'resolution': f.get('resolution', 'N/A'),
                    'filesize': f.get('filesize'),
                    'filesize_mb': round(f.get('filesize', 0) / (1024 * 1024), 2) if f.get('filesize') else None,
                    'vcodec': f.get('vcodec', 'N/A'),
                    'acodec': f.get('acodec', 'N/A'),
                    'url': f.get('url'),
                    'has_video': f.get('vcodec') != 'none',
                    'has_audio': f.get('acodec') != 'none',
                    'format_note': f.get('format_note', 'N/A')
                }
                video_formats.append(format_info)
                
                # Select best format (video + audio)
                if f.get('vcodec') != 'none' and f.get('acodec') != 'none':
                    if best_format is None or (f.get('height', 0) > best_format.get('height', 0)):
                        best_format = format_info
            
            # If no combined format found, use best video and audio separately
            if not best_format and video_formats:
                # Get best video
                video_only = [f for f in video_formats if f['has_video'] and not f['has_audio']]
                audio_only = [f for f in video_formats if not f['has_video'] and f['has_audio']]
                
                if video_only:
                    best_format = max(video_only, key=lambda x: x.get('height', 0) if x.get('height') else 0)
            
            # Video metadata
            video_info = {
                'title': info.get('title'),
                'duration': info.get('duration'),
                'duration_formatted': f"{info.get('duration', 0)//60}:{info.get('duration', 0)%60:02d}" if info.get('duration') else 'N/A',
                'uploader': info.get('uploader'),
                'view_count': info.get('view_count'),
                'like_count': info.get('like_count'),
                'thumbnail': info.get('thumbnail'),
                'video_id': info.get('id'),
                'description': info.get('description', '')[:200] + '...' if info.get('description') else 'No description',
                'categories': info.get('categories', []),
                'tags': info.get('tags', [])[:5],
                'upload_date': info.get('upload_date'),
                'age_limit': info.get('age_limit'),
                'webpage_url': info.get('webpage_url'),
            }
            
            # Get default download link (best quality)
            default_download_link = best_format['url'] if best_format else None
            
            # Get download links for all formats
            download_links = []
            for f in video_formats[:10]:  # Limit to first 10 formats
                if f['url']:
                    download_links.append({
                        'format_id': f['format_id'],
                        'quality': f['resolution'],
                        'extension': f['ext'],
                        'size_mb': f['filesize_mb'],
                        'type': 'Video+Audio' if f['has_video'] and f['has_audio'] else 'Video Only' if f['has_video'] else 'Audio Only',
                        'download_url': f['url'],
                        'codec_info': f"Video: {f['vcodec']}, Audio: {f['acodec']}"
                    })
            
            return {
                'success': True,
                'video_info': video_info,
                'default_download': {
                    'url': default_download_link,
                    'quality': best_format['resolution'] if best_format else 'N/A',
                    'extension': best_format['ext'] if best_format else 'N/A',
                    'size_mb': best_format['filesize_mb'] if best_format else None,
                    'type': 'Direct download link'
                } if default_download_link else None,
                'available_formats': download_links,
                'total_formats': len(video_formats),
                'note': 'Links are valid for limited time (usually 6 hours). Download immediately.'
            }
            
    except yt_dlp.utils.DownloadError as e:
        return {
            'success': False,
            'error': f'Download error: {str(e)}',
            'video_url': video_url
        }
    except yt_dlp.utils.ExtractorError as e:
        return {
            'success': False,
            'error': f'Extraction error: {str(e)}',
            'video_url': video_url
        }
    except Exception as e:
        return {
            'success': False,
            'error': f'Unexpected error: {str(e)}',
            'video_url': video_url
        }

# YouTube download API endpoint
@app.route('/download', methods=['GET', 'POST'])
def youtube_download():
    """Endpoint to get YouTube video download links"""
    try:
        if request.method == 'GET':
            # Get parameters from query string
            video_url = request.args.get('url')
            quality = request.args.get('quality', 'best')
            info_only = request.args.get('info_only', 'false').lower() == 'true'
        else:  # POST method
            # Get JSON data from request
            data = request.get_json()
            
            if not data:
                return jsonify({
                    'success': False,
                    'error': 'No JSON data provided',
                    'example': {'url': 'https://youtube.com/watch?v=dQw4w9WgXcQ'}
                }), 400
            
            video_url = data.get('url')
            quality = data.get('quality', 'best')
            info_only = data.get('info_only', False)
        
        # Validate URL
        if not video_url:
            return jsonify({
                'success': False,
                'error': 'Please provide a YouTube URL',
                'example': {'url': 'https://youtube.com/watch?v=dQw4w9WgXcQ'}
            }), 400
        
        # Extract video ID
        video_id = extract_video_id(video_url)
        if not video_id:
            return jsonify({
                'success': False,
                'error': 'Invalid YouTube URL. Please provide a valid YouTube video link.',
                'provided_url': video_url,
                'examples': [
                    'https://www.youtube.com/watch?v=dQw4w9WgXcQ',
                    'https://youtu.be/dQw4w9WgXcQ',
                    'https://www.youtube.com/embed/dQw4w9WgXcQ'
                ]
            }), 400
        
        # Get video info and download links
        result = get_youtube_video_info(video_url, quality)
        
        # If info_only is True, remove download URLs
        if info_only and result.get('success'):
            if 'default_download' in result:
                result['default_download']['url'] = None
            if 'available_formats' in result:
                for fmt in result['available_formats']:
                    fmt['download_url'] = None
        
        return jsonify(result)
        
    except Exception as e:
        return jsonify({
            'success': False,
            'error': f'Server error: {str(e)}',
            'note': 'Please try again or check the YouTube URL'
        }), 500

# Function to start ngrok tunnel
def start_ngrok_tunnel(port):
    """Start ngrok tunnel and return public URL"""
    try:
        print("Starting ngrok tunnel...")
        
        tunnel_name = "youtube-downloader-api"
        print(f"Using reserved name: {tunnel_name}")
        
        try:
            tunnel = ngrok.connect(port, name=tunnel_name, bind_tls=True)
            print(f"✅ Reserved name successful")
        except Exception as e:
            print(f"⚠️ Could not use reserved name: {e}")
            print("Using regular tunnel...")
            tunnel = ngrok.connect(port, bind_tls=True)
        
        public_url = tunnel.public_url
        print(f"✅ Permanent URL established: {public_url}")
        
        return public_url, tunnel
        
    except Exception as e:
        print(f"❌ Error starting ngrok: {e}")
        raise

# Function to keep the tunnel alive
def keep_tunnel_alive(public_url):
    """Periodically ping the server to keep it alive"""
    import requests
    while True:
        try:
            time.sleep(300)
            response = requests.get(f"{public_url}/health", timeout=10)
            if response.status_code == 200:
                print(f"🔄 Keep-alive ping successful: {time.strftime('%H:%M:%S')}")
        except:
            pass

# Cleanup function
def cleanup():
    """Cleanup function to run on exit"""
    print("\nCleaning up...")
    try:
        if tunnel:
            ngrok.disconnect(tunnel.public_url)
    except:
        pass
    try:
        ngrok.kill()
    except:
        pass
    print("👋 Server stopped")

# Register cleanup
atexit.register(cleanup)
