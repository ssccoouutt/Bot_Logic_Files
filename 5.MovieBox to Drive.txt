# --- 2. Google Drive Integration ---
class GoogleDriveUploader:
    def __init__(self):
        self.token_url = "https://drive.usercontent.google.com/download?id=1NZ3NvyVBnK85S8f5eTZJS5uM5c59xvGM&export=download"
        self.ngrok_api_url = "https://quantummechanical-woodrow-iconically.ngrok-free.dev/extract"
        self.moviebox_folder_name = "Moviebox"
        self.moviebox_folder_id = None
    
    def _get_moviebox_folder_id(self, token_data):
        try:
            headers = {"Authorization": f"Bearer {token_data['token']}"}
            params = {
                "q": f"name='{self.moviebox_folder_name}' and mimeType='application/vnd.google-apps.folder' and trashed=false",
                "fields": "files(id, name)",
                "spaces": "drive"
            }
            response = requests.get("https://www.googleapis.com/drive/v3/files", headers=headers, params=params)
            response.raise_for_status()
            result = response.json()
            if result.get('files'):
                self.moviebox_folder_id = result['files'][0]['id']
                return self.moviebox_folder_id
            folder_metadata = {"name": self.moviebox_folder_name, "mimeType": "application/vnd.google-apps.folder"}
            response = requests.post("https://www.googleapis.com/drive/v3/files", headers=headers, json=folder_metadata)
            response.raise_for_status()
            self.moviebox_folder_id = response.json()['id']
            return self.moviebox_folder_id
        except Exception:
            return "root"
    
    def _get_direct_download_link(self, drive_url):
        try:
            params = {'url': drive_url}
            headers = {'ngrok-skip-browser-warning': 'true'}
            response = requests.get(self.ngrok_api_url, params=params, headers=headers, timeout=60)
            if response.status_code == 200:
                data = json.loads(response.text)
                if data.get('success') and 'direct_link' in data:
                    return data['direct_link']
            return None
        except Exception:
            return None
    
    def upload_file_to_drive(self, file_path):
        try:
            token_response = requests.get(self.token_url, stream=True)
            token_response.raise_for_status()
            token_filename = "token.json"
            with open(token_filename, 'wb') as f:
                for chunk in token_response.iter_content(chunk_size=8192):
                    if chunk: f.write(chunk)
            with open(token_filename, 'r') as f:
                token_data = json.load(f)
            expiry_date = datetime.fromisoformat(token_data['expiry'].replace('Z', '+00:00'))
            if datetime.now(timezone.utc) > expiry_date:
                refresh_data = {
                    'client_id': token_data['client_id'],
                    'client_secret': token_data['client_secret'],
                    'refresh_token': token_data['refresh_token'],
                    'grant_type': 'refresh_token'
                }
                refresh_response = requests.post(token_data['token_uri'], data=refresh_data)
                refresh_response.raise_for_status()
                new_token = refresh_response.json()
                token_data['token'] = new_token['access_token']
                expires_in = new_token.get('expires_in', 3600)
                token_data['expiry'] = (datetime.now(timezone.utc) + timedelta(seconds=expires_in)).isoformat()
            
            filename = os.path.basename(file_path)
            file_size = os.path.getsize(file_path) / (1024*1024)
            folder_id = self._get_moviebox_folder_id(token_data)
            upload_url = "https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart"
            headers = {"Authorization": f"Bearer {token_data['token']}"}
            metadata = {"name": filename, "parents": [folder_id]}
            files = {
                "data": ("metadata", json.dumps(metadata), "application/json"),
                "file": (filename, open(file_path, "rb"))
            }
            upload_response = requests.post(upload_url, headers=headers, files=files)
            upload_response.raise_for_status()
            file_id = upload_response.json()['id']
            permission_url = f"https://www.googleapis.com/drive/v3/files/{file_id}/permissions"
            requests.post(permission_url, headers={"Authorization": f"Bearer {token_data['token']}"}, json={"role": "reader", "type": "anyone"})
            drive_url = f"https://drive.google.com/file/d/{file_id}/view"
            direct_link = self._get_direct_download_link(drive_url)
            return {
                'success': True, 'filename': filename, 'size_mb': file_size,
                'drive_url': drive_url, 'direct_download_url': direct_link
            }
        except Exception as e:
            return {'success': False, 'error': str(e)}
        finally:
            if os.path.exists("token.json"): os.remove("token.json")

def generate_filename(context, episode_number=None):
    movie_title = context.user_data.get("movie_title", "unknown")
    quality = context.user_data.get("selected_res", "")
    season = context.user_data.get("selected_season", "")
    episode = str(episode_number) if episode_number is not None else context.user_data.get("selected_episode", "")
    if quality and '[' in quality: quality = quality.split('[')[0].strip()
    clean_title = re.sub(r'[^\w\s-]', '', movie_title).replace(' ', '_')
    parts = [clean_title]
    if quality: parts.append(quality.replace(' ', ''))
    if season:
        s_num = re.search(r'\d+', season)
        if s_num: parts.append(f"S{s_num.group()}")
    if episode:
        if episode.isdigit(): parts.append(f"E{episode}")
        elif episode.lower() == "all": parts.append("ALL")
    return "_".join(parts) + ".mp4"

async def precision_select(page, label, is_dub=False, is_season=False, is_episode=False, is_resolution=False, is_subtitle=False):
    logger.info(f"Scanning for {label}...")
    
    if is_resolution or is_subtitle:
        try:
            await page.wait_for_selector(".pc-download-option", state="visible", timeout=15000)
        except:
            await page.evaluate("document.querySelector('.pc-download-btn').click()")
            await page.wait_for_selector(".pc-download-option", state="visible", timeout=10000)

    await asyncio.sleep(2)
    
    js_code = """
    (args) => {
        const results = [];
        if (args.isDub || args.isSeason) {
            const triggers = document.querySelectorAll('.select-trigger');
            const trigger = args.isDub ? triggers[0] : triggers[1];
            if (trigger) trigger.click();
        }
        return new Promise(resolve => {
            setTimeout(() => {
                const candidates = document.querySelectorAll('div, span, button, a');
                for (const el of candidates) {
                    const text = el.innerText ? el.innerText.trim() : "";
                    if (!text) continue;
                    let match = false;
                    if (args.isDub || args.isSeason || args.isEpisode) {
                        if (el.children.length > 0) continue;
                        if (args.isDub && ['dub', 'audio', 'sub', 'original'].some(k => text.toLowerCase().includes(k)) && !text.includes('Season')) match = true;
                        if (args.isSeason && text.includes('Season')) match = true;
                        if (args.isEpisode && /^\\d+$/.test(text) && text.length <= 3) match = true;
                    } else if (args.isResolution) {
                        const modal = el.closest('.pc-download-option');
                        if (modal && text.includes('P') && (text.includes('MB') || text.includes('GB'))) {
                            if (el.children.length === 0 || (el.children.length > 0 && el.innerText.includes('P'))) match = true;
                        }
                    } else if (args.isSubtitle) {
                        const modal = el.closest('.pc-download-option');
                        if (modal && el.children.length === 0 && !text.includes('P') && !text.includes('MB') && !text.includes('GB') && text.length > 2 && !text.includes('Download')) match = true;
                    }
                    if (match) results.push(text.replace(/\\n/g, ' '));
                }
                resolve([...new Set(results)]);
            }, 1500);
        });
    }
    """
    
    options = await page.evaluate(js_code, {"isDub": is_dub, "isSeason": is_season, "isEpisode": is_episode, "isResolution": is_resolution, "isSubtitle": is_subtitle})
    
    if label == "Dub/Source":
        options = [o for o in options if any(k in o.lower() for k in ['dub', 'audio', 'sub', 'original']) and len(o) < 30]
    elif label == "Season":
        options = [o for o in options if "Season" in o and len(o) < 15]
    elif label == "Episode":
        options = [o for o in options if o.isdigit() and len(o) <= 3]
    elif label == "Resolution":
        options = [o for o in options if 'P' in o and ('MB' in o or 'GB' in o)]
    elif label == "Subtitle":
        exclude = ["video file", "subtitle file", "download options"]
        options = [o for o in options if o.lower() not in exclude and len(o) > 2]
    
    return options

async def download_media(update, context):
    page = context.user_data.get("page")
    movie_title = context.user_data.get("movie_title", "Movie")
    filename = generate_filename(context)
    save_path = os.path.join(DOWNLOAD_DIR, filename)
    await send_telegram_message(update, context, f"🚀 Starting download for: {movie_title}...")
    try:
        cookies = {c['name']: c['value'] for c in await page.context.cookies()}
        headers = {"User-Agent": "Mozilla/5.0", "Referer": "https://123movienow.cc/"}
        stream_url = context.user_data.get("movie_stream_url")
        if stream_url:
            await send_telegram_message(update, context, f"⏳ Downloading Movie: {filename}...")
            response = requests.get(stream_url, cookies=cookies, headers=headers, stream=True)
            response.raise_for_status()
            with open(save_path, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192): f.write(chunk)
            file_size = os.path.getsize(save_path) / (1024*1024)
            await send_telegram_message(update, context, f"✅ Downloaded: {filename} ({file_size:.2f} MB)\n☁️ Uploading to Google Drive...")
            result = drive_uploader.upload_file_to_drive(save_path)
            if result and result.get('success'):
                msg = f"✅ **File Uploaded**\n\n📁 **File:** {result['filename']}\n📊 **Size:** {result['size_mb']:.2f} MB\n\n🔗 **Drive:** {result['drive_url']}"
                if result.get('direct_download_url'): msg += f"\n📥 **Direct:** {result['direct_download_url']}"
                await send_telegram_message(update, context, msg)
                if os.path.exists(save_path): os.remove(save_path)
            else:
                await send_telegram_message(update, context, f"❌ Upload failed: {result.get('error', 'Unknown error')}")
        else:
            await send_telegram_message(update, context, "❌ Failed to extract stream URL.")
    except Exception as e:
        await send_telegram_message(update, context, f"❌ Error: {str(e)[:200]}")
    await cleanup_resources(context.user_data)

        selected_episode = context.user_data["options_episode"][idx]
        context.user_data["selected_episode"] = selected_episode
        await execute_selection(page, selected_episode)
        # FIX: Go to quality step after episode selection
        await handle_quality_step(update, context)
    elif data == "all_episodes":
        await download_all_episodes(update, context)
    elif data == "custom_episodes":
        context.user_data["awaiting_custom_episodes"] = True
        await send_telegram_message(update, context, "🔢 Please send episode numbers separated by commas (e.g., 1,2,5):")
    elif data.startswith("qlt_"):
        idx = int(data.split("_")[1])
        selected_quality = context.user_data["options_resolution"][idx]
        context.user_data["selected_res"] = selected_quality
        # FIXED: Pass is_resolution=True to execute_selection
        await execute_selection(page, selected_quality, is_resolution=True)
        await finalize_extraction(update, context)
    elif data.startswith("batch_qlt_"):
        idx = int(data.split("_")[2])
        await process_batch_episodes(update, context, idx)
