def verify_chrome_installation():

    system = platform.system().lower()
    chrome_paths = []
    
    if system == "linux":
        chrome_paths = [
            "/usr/bin/google-chrome-stable",
            "/usr/bin/google-chrome",
            "/usr/bin/chromium-browser",
            "/usr/bin/chromium",
            "/usr/local/bin/google-chrome-stable",
            "/opt/google/chrome/google-chrome",
            "/snap/bin/chromium",
            "/snap/bin/chrome"
        ]
    elif system == "darwin":
        chrome_paths = [
            "/Applications/Google Chrome.app/Contents/MacOS/Google Chrome",
            "/Applications/Chromium.app/Contents/MacOS/Chromium"
        ]
    elif system == "windows":
        chrome_paths = [
            "C:\\Program Files\\Google\\Chrome\\Application\\chrome.exe",
            "C:\\Program Files (x86)\\Google\\Chrome\\Application\\chrome.exe"
        ]
    
    for path in chrome_paths:
        if os.path.exists(path):
            print(f"✓ Found Chrome at: {path}")
            return path
    
    print("✗ Chrome not found in standard locations")
    return None

def zipdir(src_dir, zipfilename):
    """Zip directory excluding lock files"""
    with zipfile.ZipFile(zipfilename, "w", zipfile.ZIP_DEFLATED) as zf:
        for root, dirs, files in os.walk(src_dir):
            for file in files:
                if (file.startswith("Singleton") or 
                    file.endswith(".lock") or 
                    file.endswith("Lock") or 
                    file.endswith("Socket")):
                    continue
                filepath = os.path.join(root, file)
                arcname = os.path.relpath(filepath, src_dir)
                try:
                    zf.write(filepath, arcname=arcname)
                except Exception as e:
                    print(f"Skipping {filepath}: {e}")

def load_profile_zip(file_path):
    """Load browser profile from zip file"""
    if os.path.exists(USER_DATA_DIR):
        shutil.rmtree(USER_DATA_DIR)
    os.makedirs(USER_DATA_DIR, exist_ok=True)
    
    with zipfile.ZipFile(file_path, "r") as zf:
        for member in zf.namelist():
            if ("/Singleton" in member or 
                member.endswith("SingletonLock") or 
                member.endswith("SingletonSocket") or 
                member.endswith(".lock")):
                continue
            try:
                zf.extract(member, USER_DATA_DIR)
            except Exception as e:
                print(f"Skipping {member}: {e}")
    
    print(f"[INFO] Browser profile extracted to {USER_DATA_DIR}")

def download_any_url(url, save_path):
    """Download file from URL"""
    response = requests.get(url, stream=True)
    response.raise_for_status()
    
    with open(save_path, "wb") as f:
        for chunk in response.iter_content(chunk_size=8192):
            f.write(chunk)

    def soft_restart(self):
        """Restart the browser"""
        self.start_driver()

    def capture_screenshot(self):
        """Capture screenshot of current page"""
        screenshot_bytes = self.driver.get_screenshot_as_png()
        img = Image.open(io.BytesIO(screenshot_bytes))
        img_bytes = io.BytesIO()
        img.save(img_bytes, format="PNG")
        img_bytes.seek(0)
        return img_bytes

    def mark_screenshot(self, screenshot_bytes, click_x=None, click_y=None):
        """Add marker to screenshot at click coordinates"""
        img = Image.open(io.BytesIO(screenshot_bytes))
        draw = ImageDraw.Draw(img)
        
        if click_x is not None and click_y is not None:
            dot_radius = 10
            draw.ellipse(
                [(click_x - dot_radius, click_y - dot_radius), 
                 (click_x + dot_radius, click_y + dot_radius)],
                fill=(255, 0, 0),
            )
            draw.ellipse(
                [(click_x - dot_radius - 2, click_y - dot_radius - 2),
                 (click_x + dot_radius + 2, click_y + dot_radius + 2)],
                outline=(255, 255, 255),
                width=2,
            )
        
        img_bytes = io.BytesIO()
        img.save(img_bytes, format="PNG")
        img_bytes.seek(0)
        return img_bytes

    def get_tab_titles(self):
        """Get list of all open tabs"""
        titles = []
        curr_handle = self.driver.current_window_handle
        handles = self.driver.window_handles
        
        for i, h in enumerate(handles):
            self.driver.switch_to.window(h)
            try:
                title = self.driver.title
            except Exception as e:
                title = f"[UNKNOWN] {str(e)}"
            
            url = ""
            try:
                url = self.driver.current_url
            except Exception:
                url = ""
            
            titles.append(f"{i}: {title}\n    {url}")
        
        self.driver.switch_to.window(curr_handle)
        return "\n".join(titles)

    def switch_to_tab(self, idx: int):
        """Switch to specific tab by index"""
        handles = self.driver.window_handles
        if 0 <= idx < len(handles):
            self.driver.switch_to.window(handles[idx])
            return f"Switched to tab {idx}: {self.driver.title}"
        else:
            return f"Tab index {idx} out of range ({len(handles)} tabs open)."

    def refresh_tab(self):
        """Refresh current tab"""
        self.driver.refresh()
        time.sleep(1)
        return f"Refreshed: {self.driver.title or self.driver.current_url}"

    def open_new_tab(self, url):
        """Open new tab with URL"""
        self.driver.execute_script(f"window.open('{url}', '_blank');")
        time.sleep(1)
        self.driver.switch_to.window(self.driver.window_handles[-1])
        return f"Opened and switched to new tab: {url}"

    def extract_text_from_brackets(self, text):
        """Extract text within square brackets"""
        if "[" in text and "]" in text:
            start, end = text.rfind("[") + 1, text.rfind("]")
            if start < end:
                return text[start:end]
        return None

    def process_action(self, action_text):
        """Process user action command - EXACTLY AS YOUR ORIGINAL"""
        try:
            if "," in action_text:
                parts = action_text.split(maxsplit=1)
                x, y = map(int, parts[0].split(","))
                text_to_type = None
                
                if len(parts) > 1:
                    bracket_text = self.extract_text_from_brackets(parts[1])
                    text_to_type = bracket_text if bracket_text else parts[1]
                
                actions = ActionChains(self.driver)
                actions.move_by_offset(x, y).click().perform()
                
                if text_to_type:
                    for char in text_to_type:
                        if char == "\n":
                            actions.key_down(Keys.SHIFT).send_keys(Keys.ENTER).key_up(Keys.SHIFT).perform()
                        else:
                            actions.send_keys(char).perform()
                        time.sleep(0.05)
                
                actions.move_by_offset(-x, -y).perform()
                msg = f"Clicked at ({x},{y})"
                if text_to_type:
                    msg += f" and typed:\n{text_to_type}"
                
                time.sleep(2)
                screenshot_bytes = self.driver.get_screenshot_as_png()
                marked = self.mark_screenshot(screenshot_bytes, x, y)
                return msg, marked
            
            return (
                "🌐 Chrome Browser Automation Bot\n\n"
                "Commands:\n"
                "- x,y [text] → Click at (x,y) and type text\n"
                "- x,y → Click at (x,y)\n"
                "- /newtab <url>   Open new tab with URL\n"
                "- /switch <tab>   Switch to tab index\n"
                "- /tabs           List open tabs\n"
                "- /save           Download browser profile\n"
                "- /load <link>    Restore profile from link\n"
                "- /refresh        Refresh current tab\n"
                "- /get            Screenshot\n"
                "- /cload <cookie link> Inject cookies and go to google.com\n"
            ), None
        
        except Exception as e:
            logger.error(f"Error processing action: {e}")
            return f"Error occurred: {str(e)}", None

                    if exp.replace(".", "", 1).isdigit():
                        c["expiry"] = int(float(exp))
                    cookies.append(c)
                except:
                    continue
        
        if not cookies:
            return await update.message.reply_text("❌ No cookies found")
        
        bot_instance.driver.get("https://google.com")
        time.sleep(2)
        bot_instance.driver.delete_all_cookies()
        
        cnt = 0
        for c in cookies:
            for k in ["sameSite", "id", "storeId", "hostOnly", "httpOnly"]:
                c.pop(k, None)
            if "expiry" in c:
                try:
                    c["expiry"] = int(c["expiry"])
                except:
                    c.pop("expiry")
            try:
                bot_instance.driver.add_cookie(c)
                cnt += 1
            except:
                pass
        
        bot_instance.driver.get("https://google.com")
        time.sleep(2)
        await update.message.reply_text(f"✅ {cnt} cookies injected and loaded Google.com")
    except Exception as e:
        await update.message.reply_text(f"❌ cload failed: {e}")
