def parse_netscape_cookies(cookie_file_path):
    """
    Parse Netscape HTTP Cookie file format to Playwright format
    """
    playwright_cookies = []
    
    try:
        with open(cookie_file_path, 'r', encoding='utf-8') as f:
            lines = f.readlines()
        
        for line in lines:
            # Skip comments and empty lines
            if line.startswith('#') or not line.strip():
                continue
            
            parts = line.strip().split('\t')
            if len(parts) < 7:
                continue
            
            domain = parts[0]
            flag = parts[1].upper() == "TRUE"
            path = parts[2]
            secure = parts[3].upper() == "TRUE"
            expiration = float(parts[4]) if parts[4] else None
            name = parts[5]
            value = parts[6]
            
            # Skip if cookie is expired
            if expiration and expiration < datetime.now().timestamp():
                continue
            
            # Create cookie object for Playwright
            cookie = {
                'name': name,
                'value': value,
                'domain': domain,
                'path': path,
                'secure': secure,
                'httpOnly': False,
                'sameSite': 'Lax',
            }
            
            # Add expiration if present
            if expiration:
                cookie['expires'] = expiration
            
            playwright_cookies.append(cookie)
        
        logger.info(f"✅ Parsed {len(playwright_cookies)} cookies from {cookie_file_path}")
        return playwright_cookies
        
    except Exception as e:
        logger.error(f"Failed to parse cookies: {e}")
        return []

def download_file_from_url(url):
    """Download file from URL with proper filename detection"""
    try:
        with requests.get(url, stream=True, allow_redirects=True) as response:
            content_disposition = response.headers.get('content-disposition')
            filename = None

            if content_disposition:
                fname = re.findall('filename=(.+)', content_disposition)
                if fname:
                    filename = fname[0].strip('"\'')

            if not filename:
                parsed = urlparse(url)
                filename = os.path.basename(parsed.path)
                if not filename:
                    filename = f"download_{int(time.time())}"
                
                # Add extension based on content type
                if '.' not in filename:
                    content_type = response.headers.get('content-type', '')
                    if 'image/jpeg' in content_type:
                        filename += '.jpg'
                    elif 'image/png' in content_type:
                        filename += '.png'
                    elif 'application/pdf' in content_type:
                        filename += '.pdf'
                    elif 'application/zip' in content_type:
                        filename += '.zip'
                    elif 'text/plain' in content_type:
                        filename += '.txt'
                    else:
                        filename += '.bin'
            
            # Clean filename
            filename = re.sub(r'[^\w\-_. ]', '_', filename)
            filepath = os.path.join(UPLOAD_TEMP_DIR, filename)
            
            with open(filepath, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    if chunk:
                        f.write(chunk)
            
            logger.info(f"✅ Downloaded file: {filename} ({os.path.getsize(filepath)} bytes)")
            return filepath
    except Exception as e:
        logger.error(f"Error downloading file from URL: {e}")
        return None

async def extract_gemini_response(page):
    """Improved extraction with smart detection of completion"""
    try:
        logger.info("⏳ Waiting for response completion...")
        
        last_text = ""
        stable_count = 0
        max_wait = 120 # 2 minutes max
        
        for i in range(max_wait):
            # 1. Check for "Stop" button or "Generating" indicators using standard JS
            is_generating = await page.evaluate('''() => {
                // Check for stop button by aria-label or text content
                const buttons = Array.from(document.querySelectorAll('button'));
                const stopBtn = buttons.find(b => {
                    const label = (b.getAttribute('aria-label') || '').toLowerCase();
                    const text = (b.innerText || '').toLowerCase();
                    return label.includes('stop') || text.includes('stop');
                });
                
                if (stopBtn && stopBtn.offsetParent !== null) return true;
                
                // Check for progress bars or pulsing animations
                const progress = document.querySelector('mat-progress-bar, .generating, .loading, .typing-indicator');
                if (progress && progress.offsetParent !== null) return true;
                
                return false;
            }''')
            
            # 2. Get current text from the last response
            current_text = await page.evaluate('''() => {
                const selectors = ['model-response', '.model-response', '[data-message-author-role="model"]', '.markdown'];
                let lastResponse = null;
                
                for (const selector of selectors) {
                    const elements = document.querySelectorAll(selector);
                    if (elements.length > 0) {
                        lastResponse = elements[elements.length - 1];
                        break;
                    }
                }
                
                return lastResponse ? lastResponse.innerText.trim() : "";
            }''')
            
            if current_text:
                # If text is non-empty and has changed, reset stability
                if current_text != last_text:
                    stable_count = 0
                    last_text = current_text
                elif not is_generating:
                    # Text is same as last time and no generating indicator
                    stable_count += 1
                    # If text is stable for 3 seconds, it's likely done
                    if stable_count >= 3:
                        logger.info(f"✅ Response stable and complete ({len(current_text)} chars)")
                        return current_text
            
            await asyncio.sleep(1)
            if i % 10 == 0:
                logger.info(f"Waiting... (elapsed: {i}s, generating: {is_generating}, stable: {stable_count})")
                
        if last_text:
            logger.warning("⚠️ Response extraction reached timeout but has partial text")
            return last_text
            
        logger.warning("⚠️ Response timeout")
        return "Response timeout - check screenshot"
        
    except Exception as e:
        logger.error(f"Response extraction failed: {e}")
        return f"Error extracting response: {str(e)}"

async def handle_document(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle document uploads"""
    if not GOOGLE_COOKIES:
        await update.message.reply_text("❌ No cookies loaded.")
        return
    doc = update.message.document
    file = await context.bot.get_file(doc.file_id)
    if not os.path.exists(UPLOAD_TEMP_DIR):
        os.makedirs(UPLOAD_TEMP_DIR)
    safe_filename = re.sub(r'[^\w\-_. ]', '_', doc.file_name or f"document_{doc.file_id}")
    file_path = os.path.join(UPLOAD_TEMP_DIR, safe_filename)
    await file.download_to_drive(file_path)
    context.user_data['file_path'] = file_path
    context.user_data['file_url'] = f"telegram_document_{doc.file_id}"
    await update.message.reply_text(f"✅ Received file: {doc.file_name}\nNow send your query:")

async def process_gemini_request(update: Update, context: ContextTypes.DEFAULT_TYPE, query, file_path):
    """Process Gemini request"""
    if not GOOGLE_COOKIES:
        await update.message.reply_text("❌ No cookies loaded.")
        return
    msg = await update.message.reply_text("🧠 Gemini is thinking...")
    try:
        text, screenshot, error = await capture_authenticated_page("https://gemini.google.com", query, file_path)
        if error:
            await msg.edit_text(f"❌ Error: {error}")
        else:
            if text and len(text) > 10:
                if len(text) > 4000:
                    text = text[:4000] + "...\n\n📷 Full response in screenshot"
                await update.message.reply_text(f"*Gemini Response:*\n\n{text}", parse_mode='Markdown')
            if screenshot and screenshot.getbuffer().nbytes > 0:
                await update.message.reply_photo(photo=screenshot, caption=f"📸 Gemini Screenshot")
            await msg.delete()
    except Exception as e:
        logger.error(f"Gemini error: {e}", exc_info=True)
        await msg.edit_text(f"❌ Error: {str(e)[:200]}")
    finally:
        if file_path and os.path.exists(file_path):
            try: os.remove(file_path)
            except: pass

async def error_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle errors"""
    logger.error(f"Update {update} caused error: {context.error}")

def main():
    """Main function to run the bot"""
    if not os.path.exists(UPLOAD_TEMP_DIR):
        os.makedirs(UPLOAD_TEMP_DIR)
    application = Application.builder().token(TOKEN).build()
    application.add_handler(CommandHandler("start", start))
    application.add_handler(CommandHandler("status", status_command))
    application.add_handler(CommandHandler("refresh", refresh_cookies))
    application.add_handler(CommandHandler("google", google_handler))
    application.add_handler(CommandHandler("gemini", gemini_handler))
    application.add_handler(CommandHandler("query", query_command))
    application.add_handler(CommandHandler("skip", skip_command))
    application.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_text_message))
    application.add_handler(MessageHandler(filters.Document.ALL, handle_document))
    application.add_error_handler(error_handler)
    print("🚀 Google/Gemini Bot Starting...")
    application.run_polling(drop_pending_updates=True)

if __name__ == "__main__":
    main()
