# ===== PROGRESS BAR CLASS =====
class ProgressBar:
    def __init__(self, total: int, description: str = ""):
        self.total = total
        self.current = 0
        self.start_time = time.time()
        self.description = description
        self.last_update = 0
        self.last_update_time = 0
    
    def update(self, n: int = 1):
        self.current += n
    
    def get_progress_str(self, current_file: str = "") -> str:
        """Get formatted progress bar string"""
        if self.total == 0:
            percentage = 0
        else:
            percentage = (self.current / self.total) * 100
        
        elapsed = time.time() - self.start_time
        
        # Calculate ETA
        if self.current > 0 and self.total > 0:
            eta = (elapsed / self.current) * (self.total - self.current)
            eta_str = f"⏱️ **ETA:** {self.format_time(eta)}"
        else:
            eta_str = "⏱️ **ETA:** Calculating..."
        
        # Create progress bar
        bar_length = 20
        if self.total > 0:
            filled_length = int(bar_length * self.current / self.total)
        else:
            filled_length = 0
        bar = "█" * filled_length + "░" * (bar_length - filled_length)
        
        # Format sizes
        if self.total > 1024*1024*1024:  # If in GB range
            current_str = f"{self.current/(1024*1024*1024):.2f}GB"
            total_str = f"{self.total/(1024*1024*1024):.2f}GB"
            speed = self.current / elapsed / (1024*1024) if elapsed > 0 else 0
            speed_str = f"{speed:.1f} MB/s"
        elif self.total > 1024*1024:  # If in MB range
            current_str = f"{self.current/(1024*1024):.1f}MB"
            total_str = f"{self.total/(1024*1024):.1f}MB"
            speed = self.current / elapsed / (1024*1024) if elapsed > 0 else 0
            speed_str = f"{speed:.1f} MB/s"
        elif self.total > 1024:  # If in KB range
            current_str = f"{self.current/1024:.1f}KB"
            total_str = f"{self.total/1024:.1f}KB"
            speed = self.current / elapsed / 1024 if elapsed > 0 else 0
            speed_str = f"{speed:.1f} KB/s"
        else:
            current_str = f"{self.current}B"
            total_str = f"{self.total}B"
            speed = self.current / elapsed if elapsed > 0 else 0
            speed_str = f"{speed:.1f} B/s"
        
        result = (
            f"`{bar}` **{percentage:.1f}%**\n"
            f"📊 **Progress:** `{current_str}` / `{total_str}`\n"
            f"⚡ **Speed:** `{speed_str}`\n"
            f"{eta_str}"
        )
        
        if current_file:
            short_file = current_file[:40] + "..." if len(current_file) > 40 else current_file
            result += f"\n📄 **Current:** `{short_file}`"
        
        return result
    
    @staticmethod
    def format_time(seconds: float) -> str:
        """Format seconds to human readable time"""
        if seconds < 60:
            return f"{seconds:.0f}s"
        elif seconds < 3600:
            return f"{seconds//60:.0f}m {seconds%60:.0f}s"
        else:
            hours = seconds // 3600
            minutes = (seconds % 3600) // 60
            return f"{hours:.0f}h {minutes:.0f}m"

def format_size(bytes_size: int) -> str:
    """Format bytes to human readable size"""
    if bytes_size >= 1024*1024*1024:
        return f"{bytes_size/(1024*1024*1024):.2f} GB"
    elif bytes_size >= 1024*1024:
        return f"{bytes_size/(1024*1024):.2f} MB"
    elif bytes_size >= 1024:
        return f"{bytes_size/1024:.2f} KB"
    else:
        return f"{bytes_size} bytes"

def format_time_duration(seconds: float) -> str:
    """Format seconds to human readable duration"""
    if seconds < 60:
        return f"{seconds:.1f}s"
    elif seconds < 3600:
        minutes = seconds // 60
        seconds = seconds % 60
        return f"{minutes:.0f}m {seconds:.0f}s"
    else:
        hours = seconds // 3600
        minutes = (seconds % 3600) // 60
        return f"{hours:.0f}h {minutes:.0f}m"

def get_file_extension(filename: str) -> str:
    """Get file extension in lowercase"""
    return os.path.splitext(filename)[1].lower()

def is_archive_file(filename: str) -> bool:

    supported_formats = ['.zip']
    
    if rarfile_available:

def sanitize_filename(filename: str) -> str:
    """Sanitize filename for Telegram/Google Drive"""
    # Replace problematic characters
    filename = re.sub(r'[<>:"/\\|?*]', '_', filename)
    # Limit length
    if len(filename) > 200:
        name, ext = os.path.splitext(filename)
        filename = name[:200-len(ext)] + ext
    return filename

def extract_google_drive_id(url: str) -> Optional[str]:
    """Extract Google Drive file ID from URL"""
    patterns = [
        r'/file/d/([a-zA-Z0-9_-]+)',
        r'id=([a-zA-Z0-9_-]+)',
        r'/open\?id=([a-zA-Z0-9_-]+)',
    ]
    
    for pattern in patterns:
        match = re.search(pattern, url)
        if match:
            return match.group(1)
    
    return None

async def download_from_google_drive(url: str, destination: str, progress_msg=None) -> Tuple[bool, str, int]:
    """Download file from Google Drive"""
    try:
        # Extract file ID
        file_id = extract_google_drive_id(url)
        if not file_id:
            return False, "Invalid Google Drive URL", 0
        
        # Check if it's a direct download URL
        if 'export=download' in url:
            download_url = url
        else:
            download_url = f"https://drive.google.com/uc?export=download&id={file_id}"
        
        # Start download
        if progress_msg:
            await progress_msg.edit(
                f"🌐 **Downloading from Google Drive**\n"
                f"📄 Extracting file...",
                parse_mode='markdown'
            )
        
        session = requests.Session()
        response = session.get(download_url, stream=True, timeout=30)
        
        # Handle Google Drive virus scan warning
        if 'confirm=' in response.url:
            # Extract confirm token
            for key, value in response.cookies.items():
                if key.startswith('download_warning'):
                    confirm_token = value
                    download_url = f"https://drive.google.com/uc?export=download&confirm={confirm_token}&id={file_id}"
                    response = session.get(download_url, stream=True, timeout=30)
                    break
        
        response.raise_for_status()
        
        # Get file size
        file_size = int(response.headers.get('content-length', 0))
        
        # Start downloading with progress
        progress_bar = ProgressBar(file_size, "Downloading from Google Drive")
        downloaded = 0
        
        with open(destination, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                if chunk:
                    f.write(chunk)
                    downloaded += len(chunk)
                    progress_bar.current = downloaded
                    
                    # Update progress every 500KB or 1 second
                    if progress_msg and (downloaded % (512*1024) == 0 or time.time() - progress_bar.last_update > 1):
                        progress_bar.last_update = time.time()
                        try:
                            await progress_msg.edit(
                                f"🌐 **Downloading from Google Drive**\n"
                                f"{progress_bar.get_progress_str()}"
                            )
                        except:
                            pass
        
        # Verify download
        if os.path.exists(destination) and os.path.getsize(destination) > 0:
            actual_size = os.path.getsize(destination)
            return True, f"Downloaded successfully", actual_size
        else:
            return False, "Download failed - file is empty", 0
            
    except Exception as e:
        log_error(f"Google Drive download error: {e}")
        return False, f"Google Drive error: {str(e)}", 0

# ===== UNZIP/EXTRACTION FUNCTIONS =====
def extract_zip_file(zip_path: str, extract_dir: str, password: str = None) -> Tuple[bool, str, List[str]]:
    """Extract ZIP file"""
    try:
        extracted_files = []
        
        with zipfile.ZipFile(zip_path, 'r') as zip_ref:
            # Check if password is needed
            if password:
                zip_ref.setpassword(password.encode())
            
            # Get file list
            file_list = zip_ref.namelist()
            
            # Extract all files
            for file in file_list:
                try:
                    zip_ref.extract(file, extract_dir)
                    extracted_files.append(file)
                except RuntimeError as e:
                    if 'encrypted' in str(e):
                        return False, "Password required or incorrect password", []
                    else:
                        raise
        
        return True, f"Extracted {len(extracted_files)} files", extracted_files
        
    except zipfile.BadZipFile:
        return False, "Invalid or corrupted ZIP file", []
    except RuntimeError as e:
        if 'encrypted' in str(e):
            return False, "Password required or incorrect password", []
        else:
            return False, f"Extraction error: {str(e)}", []
    except Exception as e:
        return False, f"Extraction error: {str(e)}", []

        progress_bar = ProgressBar(file_size, "Uploading")

            nonlocal last_update_time
            progress_bar.total = total
            progress_bar.current = current

                last_update_time = current_time

                    part_size_kb=512,  # Use smaller parts for better progress
                    workers=2  # Limit workers
                ),
                timeout=3600  # 1 hour timeout for upload
            )
        except asyncio.TimeoutError:
            log_error(f"Upload timeout for: {filename}")
            await progress_msg.edit("❌ **Upload Timeout**\nThe upload took too long.")
            return False
        
        elapsed = time.time() - start_time
        speed = file_size / elapsed if elapsed > 0 else 0
        
        # Update with final stats
        await progress_msg.edit(

async def create_password_zip(files: List[Dict], zip_path: str, password: str = None, progress_msg=None) -> Tuple[bool, str]:
    """Create password protected ZIP archive with progress"""
    try:
        total_files = len(files)
        log_info(f"Creating ZIP with {total_files} files (Password: {'Yes' if password else 'No'})")
        
        if password:
            # Try to use pyminizip for better password protection
            try:
                if pyminizip_available:
                    file_list = []
                    arcname_list = []
                    
                    for idx, file_info in enumerate(files, 1):
                        file_path = file_info["path"]
                        if os.path.exists(file_path):
                            file_list.append(file_path)
                            arcname_list.append(file_info["name"])
                            
                            # Update progress
                            if progress_msg:
                                progress = (idx / total_files) * 100
                                bar_length = 20
                                filled = int(bar_length * idx / total_files)
                                bar = "█" * filled + "░" * (bar_length - filled)
                                
                                try:
                                    await progress_msg.edit(
                                        f"📦 **CREATING PASSWORD ZIP**\n\n"
                                        f"📊 **Progress:** `{idx}/{total_files}` files\n"
                                        f"🔐 **Password:** `{'*' * len(password)}`\n"
                                        f"⚡ **Processing files...**\n\n"
                                        f"`{bar}` **{progress:.1f}%**"
                                    )
                                except:
                                    pass
                    
                    # Create the password protected zip in thread pool to avoid blocking
                    loop = asyncio.get_event_loop()
                    await loop.run_in_executor(
                        executor,
                        lambda: pyminizip.compress_multiple(
                            file_list,
                            arcname_list,
                            zip_path,
                            password,
                            5  # Compression level
                        )
                    )
                    
                    log_success(f"Password ZIP created with pyminizip: {os.path.basename(zip_path)}")
                    return True, f"✅ Created password protected ZIP with {total_files} files"
                    
            except Exception as e:
                log_warning(f"pyminizip failed, falling back to basic ZIP: {e}")
                # Fallback to regular zip with password in filename
                with zipfile.ZipFile(zip_path, 'w', zipfile.ZIP_DEFLATED) as zipf:
                    for idx, file_info in enumerate(files, 1):
                        file_path = file_info["path"]
                        if os.path.exists(file_path):
                            arcname = file_info["name"]
                            zipf.write(file_path, arcname)
                            
                            # Update progress
                            if progress_msg:
                                progress = (idx / total_files) * 100
                                bar_length = 20
                                filled = int(bar_length * idx / total_files)
                                bar = "█" * filled + "░" * (bar_length - filled)
                                
                                try:
                                    await progress_msg.edit(
                                        f"📦 **CREATING PASSWORD ZIP**\n\n"
                                        f"📊 **Progress:** `{idx}/{total_files}` files\n"
                                        f"🔐 **Password:** `{'*' * len(password)}`\n"
                                        f"⚡ **Processing files...**\n\n"
                                        f"`{bar}` **{progress:.1f}%**"
                                    )
                                except:
                                    pass
                
                # Rename to indicate password
                new_path = zip_path.replace('.zip', f'_password_{password}.zip')
                os.rename(zip_path, new_path)
                return True, f"✅ Created ZIP archive with {total_files} files\n🔐 Password note: Basic protection - password is '{password}'"
        
        else:
            # Create regular ZIP
            with zipfile.ZipFile(zip_path, 'w', zipfile.ZIP_DEFLATED) as zipf:
                for idx, file_info in enumerate(files, 1):
                    file_path = file_info["path"]
                    if os.path.exists(file_path):
                        arcname = file_info["name"]
                        zipf.write(file_path, arcname)
                        
                        # Update progress
                        if progress_msg:
                            progress = (idx / total_files) * 100
                            bar_length = 20
                            filled = int(bar_length * idx / total_files)
                            bar = "█" * filled + "░" * (bar_length - filled)
                            
                            try:
                                await progress_msg.edit(
                                    f"📦 **CREATING ZIP ARCHIVE**\n\n"
                                    f"📊 **Progress:** `{idx}/{total_files}` files\n"
                                    f"⚡ **Processing files...**\n\n"
                                    f"`{bar}` **{progress:.1f}%**"
                                )
                            except:
                                pass
            
            log_success(f"Regular ZIP created: {os.path.basename(zip_path)} with {total_files} files")
            return True, f"✅ Created ZIP archive with {total_files} files"
            
    except Exception as e:
        log_error(f"ZIP creation error: {e}")
        return False, f"❌ ZIP creation error: {str(e)}"

# ===== EXTRACTION HANDLER =====
async def perform_extraction(event, archive_path: str, password: str = None):
    """Perform archive extraction"""
    user_id = event.sender_id
    session = user_sessions[user_id]
    
    # Create extraction directory
    extract_dir = os.path.join(session["temp_dir"], "extracted")
    os.makedirs(extract_dir, exist_ok=True)
    
    # Start extraction message
    msg = await event.reply(
        f"📦 **EXTRACTING ARCHIVE**\n\n"
        f"📄 `{os.path.basename(archive_path)}`\n"
        f"🔐 **Password:** `{'***' if password else 'No password'}`\n"
        f"⏳ Starting extraction...",
        parse_mode='markdown'
    )
    
    try:
        # Run extraction in thread pool
        loop = asyncio.get_event_loop()
        success, message, extracted_files = await loop.run_in_executor(
            executor,
            lambda: extract_archive(archive_path, extract_dir, password)
        )
        
        if not success:
            await msg.edit(f"❌ **EXTRACTION FAILED**\n{message}")
            return False
        
        # Count extracted files
        total_files = len(extracted_files)
        
        await msg.edit(
            f"✅ **EXTRACTION COMPLETE**\n\n"
            f"📄 **Archive:** `{os.path.basename(archive_path)}`\n"
            f"📊 **Files extracted:** `{total_files}`\n"
            f"📁 **Extracted to:** `{extract_dir}`\n"
            f"⏳ **Preparing files for upload...**",
            parse_mode='markdown'
        )
        
        # Check if there are too many files
        if total_files > 50:
            await msg.edit(
                f"📦 **TOO MANY FILES**\n\n"
                f"Extracted `{total_files}` files.\n"
                f"Too many to send individually.\n\n"
                f"💡 **Solution:**\n"
                f"1. I'll create a new ZIP with extracted files\n"
                f"2. Then send you the combined archive",
                parse_mode='markdown'
            )
            
            # Create new zip of extracted files
            output_zip = os.path.join(session["temp_dir"], "extracted_files.zip")
            files_to_zip = []
            
            # Walk through extracted directory
            for root, dirs, files in os.walk(extract_dir):
                for file in files:
                    file_path = os.path.join(root, file)
                    rel_path = os.path.relpath(file_path, extract_dir)
                    files_to_zip.append({
                        "name": rel_path,
                        "path": file_path,
                        "size": os.path.getsize(file_path)
                    })
            
            # Create new zip
            success, zip_message = await create_password_zip(files_to_zip, output_zip, None, msg)
            
            if success:
                # Send the new zip
                await msg.edit(
                    f"📦 **CREATED COMBINED ZIP**\n\n"
                    f"📄 **File:** `extracted_files.zip`\n"
                    f"📊 **Contains:** `{total_files}` files\n"
                    f"⏳ **Uploading...**",
                    parse_mode='markdown'
                )
                
                caption = f"📦 **Extracted Archive Contents**\n\n" \
                         f"📄 **Original:** `{os.path.basename(archive_path)}`\n" \
                         f"📊 **Files:** `{total_files}`\n" \
                         f"🔐 **Password:** `{'Used' if password else 'None'}`"
                
                if await send_file_with_progress(event, output_zip, caption):
                    await event.reply(
                        f"🎉 **EXTRACTION COMPLETE**\n\n"
                        f"✅ **All files have been extracted and sent!**\n\n"
                        f"📊 **Statistics:**\n"
                        f"• 📦 Original archive: `{os.path.basename(archive_path)}`\n"
                        f"• 📊 Files extracted: `{total_files}`\n"
                        f"• ⏱️ Total time: `{format_time_duration(time.time() - session['start_time'])}`\n"
                        f"• 🔐 **Password:** `{'Used' if password else 'None'}`\n\n"
                        f"🔧 *Ready for next operation*",
                        parse_mode='markdown'
                    )
                    return True
            return False
        
        # Send files individually (if less than 50)
        sent_count = 0
        failed_count = 0
        
        for idx, rel_file in enumerate(extracted_files[:50], 1):
            file_path = os.path.join(extract_dir, rel_file)
            
            if os.path.isfile(file_path):
                file_size = os.path.getsize(file_path)
                
                # Skip files larger than 2GB
                if file_size > 2 * 1024 * 1024 * 1024:
                    log_warning(f"Skipping large file: {rel_file} ({format_size(file_size)})")
                    continue
                
                try:
                    caption = f"📦 **Extracted File**\n\n" \
                             f"📄 `{rel_file}`\n" \
                             f"📊 Size: `{format_size(file_size)}`\n" \
                             f"#️⃣ {idx}/{min(total_files, 50)}"
                    
                    await msg.edit(
                        f"📤 **UPLOADING EXTRACTED FILES**\n\n"
                        f"📊 **Progress:** `{idx}/{min(total_files, 50)}`\n"
                        f"📄 **Current:** `{rel_file[:40]}...`\n"
                        f"⏳ Please wait...",
                        parse_mode='markdown'
                    )
                    
                    await client.send_file(
                        event.chat_id,
                        file_path,
                        caption=caption,
                        force_document=True
                    )
                    sent_count += 1
                    
                except Exception as e:
                    log_error(f"Failed to send {rel_file}: {e}")
                    failed_count += 1
        
        # Final message
        await event.reply(
            f"🎉 **EXTRACTION COMPLETE**\n\n"
            f"✅ **Extraction finished successfully!**\n\n"
            f"📊 **Statistics:**\n"
            f"• 📦 Archive: `{os.path.basename(archive_path)}`\n"
            f"• 📊 Total files: `{total_files}`\n"
            f"• 📤 Sent successfully: `{sent_count}`\n"
            f"• ❌ Failed to send: `{failed_count}`\n"
            f"• ⏱️ Total time: `{format_time_duration(time.time() - session['start_time'])}`\n"
            f"• 🔐 **Password:** `{'Used' if password else 'None'}`\n\n"
            f"💡 **Note:** Only first 50 files are sent\n"
            f"For more files, use the combined ZIP feature.\n\n"
            f"🔧 *Ready for next operation*",
            parse_mode='markdown'
        )
        return True
        
    except Exception as e:
        log_error(f"Extraction error: {e}")
        await event.reply(f"❌ **EXTRACTION ERROR**\n```{str(e)[:200]}```")
        return False

@client.on(events.NewMessage())
async def handle_message(event):
    """Handle all incoming messages"""
    user_id = event.sender_id
    username = event.sender.username or "Unknown"
    text = event.text.strip() if event.text else ""
    
    # Ignore commands
    if text and text.startswith('/'):
        return
    
    # Check if user is in a session
    if user_id not in user_sessions:
        # Check if user sent a file without command
        if event.media or event.document or event.photo:
            log_bot_activity(user_id, username, "sent file without command")
            await process_file_for_extraction(event)
        return
    
    session = user_sessions[user_id]
    mode = session.get("mode")
    
    # Handle password input for zip_password mode
    if mode == "zip_password" and session.get("waiting_password"):
        if not text or len(text) < 4:
            await event.reply(
                "❌ **PASSWORD TOO SHORT**\n"
                "Minimum 4 characters required.\n"
                "Please enter a stronger password:",
                parse_mode='markdown'
            )
            return
        
        log_bot_activity(user_id, username, f"setting password: {'*' * len(text)}")
        
        # Set password and create temp directory
        session["password"] = text
        session["waiting_password"] = False
        session["waiting_files"] = True
        session["temp_dir"] = tempfile.mkdtemp(prefix=f"zip_pw_{user_id}_")
        
        await event.reply(
            f"✅ **PASSWORD SET SUCCESSFULLY**\n\n"
            f"🔐 **Password:** `{'*' * len(text)}`\n\n"
            f"📥 **Now send me files to add:**\n"
            "• Telegram files (any type)\n"
            "• Direct download links\n"
            "• Google Drive links\n\n"
            f"📊 **Current:** 0 files, 0 MB\n\n"
            f"⚡ **Commands:**\n"
            f"• `/done` - Create archive\n"
            f"• `/list` - View files\n"
            f"• `/cancel` - Cancel",
            parse_mode='markdown'
        )
        return
    
    # Handle password input for encrypted archive extraction
    if mode == "extract" and session.get("password_asked"):
        log_bot_activity(user_id, username, f"entered password for extraction: {'*' * len(text)}")
        await perform_extraction(event, session["archive_path"], text)
        # Cleanup session after extraction
        asyncio.create_task(cleanup_user_session(user_id))
        return
    
    # Handle URL input for extract mode
    if mode == "extract" and session.get("waiting_file") and text:
        if text.startswith(('http://', 'https://')):
            log_bot_activity(user_id, username, f"sent URL for extraction: {text[:50]}...")
            await process_url_for_extraction(event, text)
        return
    
    # Handle URL input for zip modes
    if mode in ["zip", "zip_password"] and session.get("waiting_files") and text:
        if text.startswith(('http://', 'https://')):
            log_bot_activity(user_id, username, f"sent URL for zip: {text[:50]}...")
            await add_url_to_zip(event, text)
        return
    
    # Handle file attachments
    if event.media or event.document or event.photo:
        if mode == "extract" and session.get("waiting_file"):
            log_bot_activity(user_id, username, "sent archive file for extraction")
            await process_file_for_extraction(event)
        elif mode in ["zip", "zip_password"] and session.get("waiting_files"):
            log_bot_activity(user_id, username, "sent file to add to zip")
            await add_file_to_zip(event)
        return

async def download_from_url(url: str, destination: str, progress_msg=None) -> Tuple[bool, str, int]:
    """Download file from regular URL"""
    try:
        response = requests.get(url, stream=True, timeout=300)
        response.raise_for_status()
        
        file_size = int(response.headers.get('content-length', 0))
        downloaded = 0
        
        progress_bar = ProgressBar(file_size, "Downloading")
        
        with open(destination, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                if chunk:
                    f.write(chunk)
                    downloaded += len(chunk)
                    progress_bar.current = downloaded
                    
                    # Update progress every 500KB or 1 second
                    if progress_msg and (downloaded % (512*1024) == 0 or time.time() - progress_bar.last_update > 1):
                        progress_bar.last_update = time.time()
                        try:
                            await progress_msg.edit(
                                f"📥 **DOWNLOADING FROM URL**\n"
                                f"{progress_bar.get_progress_str()}"
                            )
                        except:
                            pass
        
        # Verify download
        if os.path.exists(destination) and os.path.getsize(destination) > 0:
            actual_size = os.path.getsize(destination)
            return True, f"Downloaded successfully", actual_size
        else:
            return False, "Download failed - file is empty", 0
            
    except Exception as e:
        log_error(f"URL download error: {e}")
        return False, f"Download error: {str(e)}", 0

# ===== EXTRACTION PROCESSING FUNCTIONS =====
async def process_file_for_extraction(event):
    """Process Telegram file for extraction"""
    user_id = event.sender_id
    
    # If user not in session, create one
    if user_id not in user_sessions:
        user_sessions[user_id] = {
            "mode": "extract",
            "waiting_file": True,
            "password_asked": False,
            "archive_path": None,
            "temp_dir": None,
            "start_time": time.time(),
            "username": event.sender.username or "Unknown"
        }
    
    session = user_sessions[user_id]
    
    # Check if file is an archive
    filename = "file"
    if event.media and hasattr(event.media, 'document'):
        doc = event.media.document
        for attr in doc.attributes:
            if isinstance(attr, DocumentAttributeFilename):
                filename = attr.file_name
                break
    
    if not is_archive_file(filename):
        await event.reply(
            f"❌ **NOT AN ARCHIVE FILE**\n"
            f"File: `{filename}`\n\n"
            f"📁 **Supported formats:**\n"
            f"• ZIP (.zip)\n"
            f"{'• RAR (.rar)' if rarfile_available else '• ⚠️ RAR (.rar) - install rarfile'}\n"
            f"{'• 7Z (.7z)' if py7zr_available else '• ⚠️ 7Z (.7z) - install py7zr'}\n\n"
            f"Please send a valid archive file.",
            parse_mode='markdown'
        )
        return
    
    # Create temp directory
    if not session.get("temp_dir"):
        session["temp_dir"] = tempfile.mkdtemp(prefix=f"extract_{user_id}_")
    
    # Download the file
    archive_path = os.path.join(session["temp_dir"], filename)
    
    msg = await event.reply(
        f"📥 **DOWNLOADING ARCHIVE**\n\n"
        f"📄 `{filename}`\n"
        f"⏳ Preparing download...",
        parse_mode='markdown'
    )
    
    success, message = await download_telegram_file(event, archive_path)
    
    if not success:
        shutil.rmtree(session["temp_dir"], ignore_errors=True)
        await msg.edit(f"❌ **DOWNLOAD FAILED**\n{message}")
        return
    
    # Update session
    session["archive_path"] = archive_path
    session["waiting_file"] = False
    
    # Check if archive needs password
    try:
        ext = get_file_extension(filename)
        if ext == '.zip':
            with zipfile.ZipFile(archive_path, 'r') as zip_ref:
                # Try to read file list without password
                file_list = zip_ref.namelist()
                if not file_list:
                    session["password_asked"] = True
                    await msg.edit(
                        f"🔐 **PASSWORD REQUIRED**\n\n"
                        f"📄 `{filename}`\n"
                        f"📊 **Size:** `{format_size(os.path.getsize(archive_path))}`\n\n"
                        f"Please enter the password for this archive:",
                        parse_mode='markdown'
                    )
                    return
        elif ext == '.rar' and rarfile_available:
            with rarfile.RarFile(archive_path, 'r') as rar_ref:
                try:
                    file_list = rar_ref.namelist()
                except rarfile.PasswordRequired:
                    session["password_asked"] = True
                    await msg.edit(
                        f"🔐 **PASSWORD REQUIRED**\n\n"
                        f"📄 `{filename}`\n"
                        f"📊 **Size:** `{format_size(os.path.getsize(archive_path))}`\n\n"
                        f"Please enter the password for this archive:",
                        parse_mode='markdown'
                    )
                    return
        elif ext == '.7z' and py7zr_available:
            # Try without password first
            try:
                with py7zr.SevenZipFile(archive_path, 'r') as sevenz_ref:
                    file_list = sevenz_ref.getnames()
            except Exception:
                session["password_asked"] = True
                await msg.edit(
                    f"🔐 **PASSWORD MAY BE REQUIRED**\n\n"
                    f"📄 `{filename}`\n"
                    f"📊 **Size:** `{format_size(os.path.getsize(archive_path))}`\n\n"
                    f"Please enter the password for this archive (if any):",
                    parse_mode='markdown'
                )
                return
    except Exception:
        # If we can't determine, assume no password
        pass
    
    # No password needed, start extraction
    await perform_extraction(event, archive_path)

async def process_url_for_extraction(event, url):

    msg = await event.reply(

    )

    if 'drive.google.com' in url and google_drive_available:
        success, message, file_size = await download_from_google_drive(url, archive_path, msg)
    else:
        # Regular URL download
        success, message, file_size = await download_from_url(url, archive_path, msg)
    
    if not success:
        await msg.edit(f"❌ **DOWNLOAD FAILED**\n{message}")
        return
    
    # Update session
    session["archive_path"] = archive_path
    session["waiting_file"] = False

            f"📄 `{filename}`\n"
            f"The downloaded file is not a supported archive format.",
            parse_mode='markdown'
        )
        return
    
    # For URL downloads, we'll ask for password if needed during extraction
    # Start extraction without password first
    await perform_extraction(event, archive_path)

# ===== CLEANUP FUNCTION =====
async def cleanup_user_session(user_id: int):
    """Cleanup user session and temp files"""
    if user_id in user_sessions:
        session = user_sessions[user_id]
        username = session.get("username", "Unknown")
        
        log_info(f"Cleaning up session for user {username}")
        
        try:
            # Remove temp files
            for file_info in session.get("files", []):
                try:
                    if os.path.exists(file_info["path"]):
                        os.remove(file_info["path"])
                except:
                    pass
            
            # Remove temp directory
            if "temp_dir" in session and session["temp_dir"]:
                try:
                    shutil.rmtree(session["temp_dir"], ignore_errors=True)
                except:
                    pass
            
            # Remove archive file
            if "archive_path" in session and session["archive_path"]:
                try:
                    if os.path.exists(session["archive_path"]):
                        os.remove(session["archive_path"])
                except:
                    pass
            
            # Remove extracted directory
            extract_dir = os.path.join(session["temp_dir"], "extracted") if session.get("temp_dir") else None
            if extract_dir and os.path.exists(extract_dir):
                try:
                    shutil.rmtree(extract_dir, ignore_errors=True)
                except:
                    pass
                    
        except Exception as e:
            log_error(f"Error cleaning up session for {username}: {e}")
        
        # Remove session
        if user_id in user_sessions:
            del user_sessions[user_id]
        
        log_success(f"Session cleaned up for user {username}")

# ===== SHUTDOWN HANDLER =====
async def shutdown():
    """Clean shutdown of the bot"""
    log_info("Shutting down bot...")
    
    # Cleanup all user sessions
    cleanup_tasks = []
    for user_id in list(user_sessions.keys()):
        cleanup_tasks.append(cleanup_user_session(user_id))
    
    if cleanup_tasks:
        await asyncio.gather(*cleanup_tasks, return_exceptions=True)
    
    # Shutdown thread pool
    executor.shutdown(wait=True)
    
    log_success("Bot shutdown complete")

    max_retries = 5
    
    while retry_count < max_retries:
        try:
            await client.start(bot_token=BOT_TOKEN)
            
            # Get bot info
            me = await client.get_me()
            log_success(f"Bot authenticated: @{me.username}")
            
            # Keep running
            await client.run_until_disconnected()
            break
            
        except (ConnectionError, TimeoutError) as e:
            retry_count += 1
            if retry_count < max_retries:
                log_warning(f"Connection failed (attempt {retry_count}/{max_retries}): {e}")
                await asyncio.sleep(5 * retry_count)  # Exponential backoff
            else:
                log_error(f"Failed to connect after {max_retries} attempts")
                raise
        except KeyboardInterrupt:
            log_info("Bot stopped by user")
            break
        except Exception as e:
            log_error(f"Unexpected error: {e}")
            break
        finally:
            await shutdown()
