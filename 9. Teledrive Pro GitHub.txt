#!/usr/bin/env python3

import os
import sys
import subprocess
import re
import time

# ========== Install Git and clone repository ==========
def install_git():
    """Install Git if not available"""
    try:
        subprocess.run(["git", "--version"], check=True, capture_output=True)
    except (subprocess.CalledProcessError, FileNotFoundError):
        print("Installing Git...")
        subprocess.run(["apt-get", "update"], check=True)
        subprocess.run(["apt-get", "install", "-y", "git"], check=True)

# ========== Clone repository exactly like original ==========
def clone_repository():
    """Clone repository exactly like !git clone"""
    print("Cloning Teledrive-Pro repository...")
    subprocess.run(["git", "clone", "https://github.com/ssccoouutt/Teledrive-Pro.git"], check=True)
    
    # Change directory exactly like %cd
    os.chdir("Teledrive-Pro")
    print(f"Changed to directory: {os.getcwd()}")

# ========== Install requirements exactly like original ==========
def install_requirements():
    """Install requirements exactly like !pip install"""
    print("Installing requirements...")
    subprocess.run([sys.executable, "-m", "pip", "install", "-r", "requirements.txt"], check=True)

# ========== Run grep exactly like original ==========
def grep_for_ports():
    """Run grep exactly like original command"""
    print("Looking for PORT/port/webhook in teledrive_pro.py...")
    try:
        # Run grep command
        result = subprocess.run(
            ["grep", "-n", "PORT\|port\|webhook", "teledrive_pro.py"], 
            capture_output=True, 
            text=True, 
            check=True
        )
        
        # Get first 10 lines exactly like head -10
        lines = result.stdout.strip().split('\n')
        for i, line in enumerate(lines[:10]):
            print(line)
    except Exception as e:
        print(f"Grep error: {e}")

# ========== Run the bot exactly like original ==========
def run_teledrive_pro():
    """Run teledrive_pro.py exactly like original"""
    print("Starting Teledrive-Pro Telegram bot...")
    
    # Check if file exists
    if not os.path.exists("teledrive_pro.py"):
        print("ERROR: teledrive_pro.py not found!")
        return
    
    # Run the bot
    try:
        subprocess.run([sys.executable, "teledrive_pro.py"], check=True)
    except KeyboardInterrupt:
        print("\nBot stopped by user")
    except Exception as e:
        print(f"Error running teledrive_pro.py: {e}")

# ========== Main execution exactly like original ==========
def main():
    """Execute exactly like the original Colab script"""
    print("="*60)
    print("TELEDRIVE-PRO - EXACT SAME AS ORIGINAL")
    print("="*60)
    
    # 1. Install Git if needed
    install_git()
    
    # 2. Clone repository (exact same as !git clone)
    clone_repository()
    
    # 3. Install requirements (exact same as !pip install -r requirements.txt)
    install_requirements()
    
    # 4. Run grep (exact same as !grep -n "PORT\|port\|webhook" teledrive_pro.py | head -10)
    grep_for_ports()
    
    # 5. Print message (exact same)
    print("Starting Teledrive-Pro Telegram bot...")
    
    # 6. Run the bot (exact same as !python teledrive_pro.py)
    run_teledrive_pro()

if __name__ == "__main__":
    main()
