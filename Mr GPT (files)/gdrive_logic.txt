def task2_upload_to_gdrive(file_url, chat_id):
    """Download from URL and upload to Google Drive - Professional Version"""
    import os, time, tempfile, logging, json, requests, urllib.parse
    from datetime import datetime, timezone
    from __main__ import send_telegram_message, logger
    
    def send_progress(text):
        send_telegram_message(chat_id, text)
    
    def download_and_refresh_token():
        """Download and refresh Google Drive token"""
        try:
            token_url = "https://drive.usercontent.google.com/download?id=1NZ3NvyVBnK85S8f5eTZJS5uM5c59xvGM&export=download"
            logger.info("[GDRIVE] Downloading token.json...")
            
            token_response = requests.get(token_url, stream=True, timeout=30)
            token_response.raise_for_status()
            
            # Extract filename
            import urllib.parse as up
            token_filename = up.unquote(os.path.basename(token_url.split('?')[0]))
            if '.' not in token_filename:
                token_filename = "token.json"
            
            # Save token locally
            temp_dir = tempfile.gettempdir()
            temp_token_path = os.path.join(temp_dir, token_filename)
            
            with open(temp_token_path, 'wb') as f:
                for chunk in token_response.iter_content(chunk_size=8192):
                    if chunk:
                        f.write(chunk)
            
            # Load token data
            with open(temp_token_path, 'r') as f:
                token_data = json.load(f)
            
            logger.info("[GDRIVE] Token data loaded successfully")
            
            # Check if token is expired
            expiry_date = datetime.fromisoformat(token_data['expiry'].replace('Z', '+00:00'))
            if datetime.now(timezone.utc) > expiry_date:
                logger.info("[GDRIVE] Token expired. Refreshing...")
                send_progress("<b>🔄 Refreshing Google Drive token...</b>")
                
                refresh_data = {
                    'client_id': token_data['client_id'],
                    'client_secret': token_data['client_secret'],
                    'refresh_token': token_data['refresh_token'],
                    'grant_type': 'refresh_token'
                }
                
                refresh_response = requests.post(token_data['token_uri'], data=refresh_data)
                refresh_response.raise_for_status()
                
                new_token = refresh_response.json()
                token_data['token'] = new_token['access_token']
                token_data['expiry'] = datetime.now(timezone.utc).isoformat()
                
                # Save refreshed token
                with open(temp_token_path, 'w') as f:
                    json.dump(token_data, f)
                
                logger.info("[GDRIVE] Token refreshed successfully")
                send_progress("<b>✅ Token refreshed!</b>")
            
            return token_data, temp_token_path
            
        except Exception as e:
            logger.error(f"[GDRIVE TOKEN] Error: {e}")
            send_progress(f"<b>❌ Token Error:</b> {str(e)[:200]}")
            return None, None
    
    def upload_to_google_drive(file_path, token_data):
        """Upload file to Google Drive using token"""
        try:
            filename = os.path.basename(file_path)
            file_size = os.path.getsize(file_path) / (1024 * 1024)
            
            send_progress(f"<b>☁️ Uploading {filename} ({file_size:.2f} MB) to Google Drive...</b>")
            logger.info(f"[GDRIVE UPLOAD] Starting upload: {filename} ({file_size:.2f} MB)")
            
            # Prepare metadata
            metadata = {
                "name": filename,
                "parents": ["root"]  # Upload to Drive root
            }
            
            # Prepare multipart upload
            files = {
                "data": ("metadata", json.dumps(metadata), "application/json"),
                "file": (filename, open(file_path, "rb"))
            }
            
            headers = {
                "Authorization": f"Bearer {token_data['token']}"
            }
            
            upload_url = "https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart"
            
            # Upload file
            upload_response = requests.post(
                upload_url,
                headers=headers,
                files=files,
                timeout=300  # 5 minute timeout for large files
            )
            
            upload_response.raise_for_status()
            
            result = upload_response.json()
            drive_url = f"https://drive.google.com/file/d/{result['id']}/view"
            
            success_msg = f"""
<b>✅ File Successfully Uploaded to Google Drive!</b>

📄 <b>File:</b> {filename}
📏 <b>Size:</b> {file_size:.2f} MB
🆔 <b>File ID:</b> {result['id']}
🔗 <b>Download Link:</b>
{drive_url}

💡 <b>Note:</b> File is now in your Google Drive
            """
            send_progress(success_msg)
            logger.info(f"[GDRIVE UPLOAD] Success! URL: {drive_url}")
            
            return drive_url
            
        except Exception as e:
            logger.error(f"[GDRIVE UPLOAD] Error: {e}")
            send_progress(f"<b>❌ Upload Failed:</b> {str(e)[:200]}")
            return None
    
    def download_file_from_url(url, output_path):
        """Download file from URL"""
        try:
            # Try to get filename from URL
            parsed = urllib.parse.urlparse(url)
            suggested_name = os.path.basename(parsed.path)
            
            # If no filename in URL, try Content-Disposition header
            if not suggested_name or '.' not in suggested_name:
                head_response = requests.head(url, allow_redirects=True)
                if 'Content-Disposition' in head_response.headers:
                    content_disp = head_response.headers['Content-Disposition']
                    if 'filename=' in content_disp:
                        suggested_name = content_disp.split('filename=')[1].strip('"\'').split(';')[0]
            
            # If still no proper name, create one
            if not suggested_name or '.' not in suggested_name:
                suggested_name = f"downloaded_file_{int(time.time())}.mp4"
            
            # Clean filename
            suggested_name = urllib.parse.unquote(suggested_name)
            
            logger.info(f"[DOWNLOAD] Downloading from: {url}")
            logger.info(f"[DOWNLOAD] Saving as: {suggested_name}")
            
            send_progress(f"<b>⬇️ Downloading file...</b>\n📄 {suggested_name}")
            
            # Download the file
            response = requests.get(url, stream=True, timeout=300)
            response.raise_for_status()
            
            total_size = int(response.headers.get('content-length', 0))
            downloaded = 0
            
            with open(output_path, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    if chunk:
                        f.write(chunk)
                        downloaded += len(chunk)
                        
                        # Update progress every 5MB
                        if total_size and downloaded % (5 * 1024 * 1024) < 8192:
                            progress = (downloaded / total_size) * 100
                            mb_downloaded = downloaded / (1024 * 1024)
                            logger.info(f"[DOWNLOAD] Progress: {progress:.1f}% ({mb_downloaded:.1f} MB)")
            
            # Rename to proper filename
            final_path = os.path.join(os.path.dirname(output_path), suggested_name)
            os.rename(output_path, final_path)
            
            file_size = os.path.getsize(final_path) / (1024 * 1024)
            logger.info(f"[DOWNLOAD] Download complete: {file_size:.2f} MB")
            send_progress(f"<b>✅ Download complete! ({file_size:.2f} MB)</b>")
            
            return final_path, suggested_name
            
        except Exception as e:
            logger.error(f"[DOWNLOAD] Error: {e}")
            send_progress(f"<b>❌ Download Failed:</b> {str(e)[:200]}")
            return None, None
    
    try:
        send_progress("<b>☁️ Starting Google Drive Upload Process</b>")
        logger.info(f"[TASK2] Starting upload for URL: {file_url[:100]}...")
        
        # Step 1: Get/refresh token
        token_data, token_path = download_and_refresh_token()
        if not token_data:
            return
        
        # Step 2: Download file
        temp_dir = tempfile.gettempdir()
        temp_file = os.path.join(temp_dir, f"temp_download_{int(time.time())}")
        
        downloaded_path, filename = download_file_from_url(file_url, temp_file)
        if not downloaded_path:
            return
        
        # Step 3: Upload to Google Drive
        drive_url = upload_to_google_drive(downloaded_path, token_data)
        
        # Cleanup
        cleanup_files = [downloaded_path]
        if token_path and os.path.exists(token_path):
            cleanup_files.append(token_path)
        
        for file_path in cleanup_files:
            if os.path.exists(file_path):
                try:
                    os.remove(file_path)
                    logger.info(f"[CLEANUP] Removed: {file_path}")
                except:
                    pass
        
        if drive_url:
            logger.info("[TASK2] Process completed successfully")
        else:
            send_progress("<b>❌ Google Drive Upload Failed</b>")
            
    except Exception as e:
        logger.error(f"[TASK2] Fatal error: {e}")
        send_progress(f"<b>❌ Fatal Error:</b> {str(e)[:200]}")
