def task4_video_editing(chat_id, operation, video1_url=None, video2_url=None, 
                       start_time=None, end_time=None, intro_video_id=None):
    """Video Editing - Final Working Version matching standalone script"""
    import os, time, tempfile, logging, re, subprocess, sys, json
    from __main__ import send_telegram_message, upload_to_drive, logger
    
    def log_progress(text, send_to_telegram=True):
        logger.info(f"[VIDEO EDIT] {text}")
        if send_to_telegram:
            send_telegram_message(chat_id, f"<b>{text}</b>")
    
    def extract_gdrive_file_id(url):
        """Extract file ID from various Google Drive URL formats"""
        patterns = [
            r'/file/d/([a-zA-Z0-9-_]+)',
            r'id=([a-zA-Z0-9-_]+)',
            r'([a-zA-Z0-9-_]{33})',
            r'([a-zA-Z0-9-_]{19})'
        ]
        
        for pattern in patterns:
            match = re.search(pattern, url)
            if match:
                fid = match.group(1)
                log_progress(f"Extracted File ID: {fid}", False)
                return fid
        raise ValueError("Could not extract file ID from URL")
    
    def download_gdrive_video(file_id, filename, description="video"):
        """Download video from Google Drive using gdown"""
        log_progress(f"⬇️ Downloading {description}...", True)
        
        # Install gdown if not available
        try:
            import gdown
        except ImportError:
            log_progress("Installing gdown...", False)
            subprocess.run([sys.executable, "-m", "pip", "install", "-q", "gdown"], 
                          capture_output=True, text=True)
            import gdown
        
        try:
            url = f"https://drive.google.com/uc?id={file_id}"
            gdown.download(url, filename, quiet=False)
            
            if os.path.exists(filename) and os.path.getsize(filename) > 0:
                file_size = os.path.getsize(filename) / (1024 * 1024)
                log_progress(f"✅ {description} downloaded: {file_size:.2f} MB", True)
                return True
            else:
                log_progress(f"❌ Failed to download {description}", True)
                return False
                
        except Exception as e:
            log_progress(f"❌ Download error: {str(e)}", True)
            return False
    
    def get_video_duration(filename):
        """Get video duration only"""
        try:
            cmd = [
                'ffprobe', '-v', 'error',
                '-show_entries', 'format=duration',
                '-of', 'default=noprint_wrappers=1:nokey=1',
                filename
            ]
            result = subprocess.run(cmd, capture_output=True, text=True, check=True)
            duration = float(result.stdout.strip())
            return duration
        except Exception as e:
            log_progress(f"⚠️ Could not get duration: {e}", False)
            return 0.0
    
    def get_video_resolution(filename):
        """Get video resolution"""
        try:
            cmd = [
                'ffprobe', '-v', 'error',
                '-select_streams', 'v:0',
                '-show_entries', 'stream=width,height',
                '-of', 'csv=s=x:p=0',
                filename
            ]
            result = subprocess.run(cmd, capture_output=True, text=True, check=True)
            width, height = result.stdout.strip().split('x')
            return int(width), int(height)
        except:
            return 1920, 1080
    
    def has_audio_stream(filename):
        """Check if video has audio stream"""
        cmd = [
            'ffprobe', '-v', 'error',
            '-select_streams', 'a',
            '-show_entries', 'stream=codec_type',
            '-of', 'csv=p=0',
            filename
        ]
        try:
            result = subprocess.run(cmd, capture_output=True, text=True)
            return 'audio' in result.stdout
        except:
            return False
    
    def time_to_seconds(time_str):
        """Convert HH:MM:SS or MM:SS or SS to seconds"""
        try:
            time_str = time_str.strip()
            
            # Handle "X to Y" format
            if 'to' in time_str:
                parts = time_str.split('to')
                time_str = parts[0].strip()
            
            time_parts = time_str.split(':')
            
            if len(time_parts) == 3:  # HH:MM:SS
                return int(time_parts[0]) * 3600 + int(time_parts[1]) * 60 + float(time_parts[2])
            elif len(time_parts) == 2:  # MM:SS
                return int(time_parts[0]) * 60 + float(time_parts[1])
            elif len(time_parts) == 1:  # SS
                return float(time_parts[0])
            else:
                raise ValueError(f"Invalid time: '{time_str}'. Use format like '5', '1:30', or '0:01:30'")
        except ValueError:
            raise ValueError(f"Invalid time: '{time_str}'. Use format like '5', '1:30', or '0:01:30'")
    
    def trim_video_fast(input_file, output_file, start_time, end_time=None):
        """Trim video using stream copy - FASTEST"""
        if end_time:
            duration = end_time - start_time
        else:
            duration = get_video_duration(input_file) - start_time
        
        cmd = [
            'ffmpeg',
            '-ss', str(start_time),      # Start time
            '-i', input_file,            # Input file
            '-t', str(duration),         # Duration
            '-c', 'copy',                # Stream copy (no re-encoding)
            '-avoid_negative_ts', 'make_zero',
            '-y',                        # Overwrite output
            output_file
        ]
        
        log_progress(f"Running trim command...", False)
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        if result.returncode != 0:
            log_progress(f"FFmpeg error: {result.stderr[:500]}", False)
            raise subprocess.CalledProcessError(result.returncode, cmd)
        
        return True
    
    def create_final_video_fast(intro_file, main_file, start_time, end_time, output_file):
        """Create final video using filter_complex - Fast and accurate"""
        
        # Get main video resolution
        main_width, main_height = get_video_resolution(main_file)
        
        log_progress(f"🎬 Creating final video...", True)
        log_progress(f"Intro: {os.path.basename(intro_file)}", True)
        log_progress(f"Main: {os.path.basename(main_file)} (trim {start_time}s to {end_time}s)", True)
        log_progress(f"Output resolution: {main_width}x{main_height}", True)
        
        # Single command: Scale intro + trim main + concatenate
        cmd = [
            'ffmpeg',
            '-i', intro_file,           # Intro video (no audio)
            '-i', main_file,            # Main video
            
            # Filter complex:
            # 1. Scale intro to match main video resolution
            # 2. Trim main video
            # 3. Add silent audio for intro
            # 4. Concatenate everything
            '-filter_complex',
            f'[0:v]scale={main_width}:{main_height}:force_original_aspect_ratio=decrease,'
            f'pad={main_width}:{main_height}:(ow-iw)/2:(oh-ih)/2,setsar=1[intro_v];'
            f'[1:v]trim=start={start_time}:end={end_time},setpts=PTS-STARTPTS[main_v];'
            f'[1:a]atrim=start={start_time}:end={end_time},asetpts=PTS-STARTPTS[main_a];'
            f'aevalsrc=0:d=5[silent];'  # 5 seconds of silence
            f'[intro_v][silent][main_v][main_a]concat=n=2:v=1:a=1[outv][outa]',
            
            '-map', '[outv]',
            '-map', '[outa]',
            
            # Fast encoding settings
            '-c:v', 'libx264',
            '-preset', 'ultrafast',
            '-crf', '23',
            '-r', '30',
            
            # Audio settings
            '-c:a', 'aac',
            '-b:a', '192k',
            
            # Ensure correct duration
            '-shortest',
            
            # Output
            '-y',
            output_file
        ]
        
        log_progress(f"🚀 Running FFmpeg...", True)
        logger.info(f"Running command: {' '.join(cmd)}")
        
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        if result.returncode != 0:
            log_progress(f"❌ FFmpeg failed!", True)
            logger.error(f"FFmpeg error: {result.stderr[:1000]}")
            return False
        
        log_progress(f"✅ Video created successfully!", True)
        return True
    
    def merge_with_intro_only(intro_file, main_file, output_file):
        """Merge intro with full main video"""
        log_progress(f"🎬 Merging intro with main video...", True)
        
        # Get main video resolution and duration
        main_width, main_height = get_video_resolution(main_file)
        main_duration = get_video_duration(main_file)
        intro_duration = get_video_duration(intro_file)
        
        log_progress(f"Intro: {intro_duration:.1f}s", True)
        log_progress(f"Main: {main_duration:.1f}s", True)
        log_progress(f"Total: {intro_duration + main_duration:.1f}s", True)
        log_progress(f"Resolution: {main_width}x{main_height}", True)
        
        cmd = [
            'ffmpeg',
            '-i', intro_file,
            '-i', main_file,
            
            '-filter_complex',
            f'[0:v]scale={main_width}:{main_height}:force_original_aspect_ratio=decrease,'
            f'pad={main_width}:{main_height}:(ow-iw)/2:(oh-ih)/2,setsar=1[intro_v];'
            f'aevalsrc=0:d={intro_duration}[silent];'
            f'[intro_v][silent][1:v][1:a]concat=n=2:v=1:a=1[outv][outa]',
            
            '-map', '[outv]',
            '-map', '[outa]',
            
            '-c:v', 'libx264',
            '-preset', 'ultrafast',
            '-crf', '23',
            '-c:a', 'aac',
            '-b:a', '192k',
            '-shortest',
            '-y',
            output_file
        ]
        
        logger.info(f"Running command: {' '.join(cmd)}")
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        if result.returncode != 0:
            log_progress(f"❌ Merge failed", True)
            logger.error(f"Merge error: {result.stderr[:500]}")
            return False
        
        log_progress(f"✅ Merge successful!", True)
        return True
    
    def merge_two_custom_videos(video1, video2, output_file):
        """Merge two custom videos"""
        log_progress(f"🔗 Merging two videos...", True)
        
        # Get durations
        v1_duration = get_video_duration(video1)
        v2_duration = get_video_duration(video2)
        
        log_progress(f"Video 1: {v1_duration:.1f} seconds", True)
        log_progress(f"Video 2: {v2_duration:.1f} seconds", True)
        log_progress(f"Total: {v1_duration + v2_duration:.1f} seconds", True)
        
        # Simple concatenation
        concat_file = os.path.join(temp_dir, f"concat_{int(time.time())}.txt")
        with open(concat_file, 'w') as f:
            f.write(f"file '{os.path.abspath(video1)}'\n")
            f.write(f"file '{os.path.abspath(video2)}'\n")
        
        # Try stream copy first
        cmd = [
            'ffmpeg',
            '-f', 'concat',
            '-safe', '0',
            '-i', concat_file,
            '-c', 'copy',
            '-movflags', '+faststart',
            '-y',
            output_file
        ]
        
        logger.info(f"Running command: {' '.join(cmd)}")
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        if result.returncode != 0:
            log_progress("⚠️ Stream copy failed, re-encoding...", True)
            cmd = [
                'ffmpeg',
                '-f', 'concat',
                '-safe', '0',
                '-i', concat_file,
                '-c:v', 'libx264',
                '-preset', 'ultrafast',
                '-crf', '23',
                '-c:a', 'aac',
                '-movflags', '+faststart',
                '-y',
                output_file
            ]
            result = subprocess.run(cmd, capture_output=True, text=True)
        
        # Clean up concat file
        if os.path.exists(concat_file):
            os.remove(concat_file)
        
        if result.returncode != 0:
            log_progress(f"❌ Merge failed", True)
            logger.error(f"Merge error: {result.stderr[:500]}")
            return False
        
        log_progress(f"✅ Merge successful!", True)
        return True
    
    temp_dir = tempfile.gettempdir()
    temp_files = []
    output_path = None
    
    try:
        if operation == 'trim':
            log_progress("✂️ TRIM VIDEO", True)
            
            # Extract file ID and download
            main_file_id = extract_gdrive_file_id(video1_url)
            main_filename = os.path.join(temp_dir, f"main_{int(time.time())}.mp4")
            temp_files.append(main_filename)
            
            if not download_gdrive_video(main_file_id, main_filename, "video"):
                return
            
            # Get video duration
            duration = get_video_duration(main_filename)
            log_progress(f"📹 Video duration: {duration:.2f} seconds ({duration/60:.1f} minutes)", True)
            
            # Validate times
            if end_time is None:
                end_time = duration
            
            if end_time <= start_time:
                log_progress(f"❌ End time ({end_time}s) must be AFTER start time ({start_time}s)", True)
                return
            
            if end_time > duration:
                log_progress(f"⚠️ End time exceeds video duration. Using video end ({duration}s)", True)
                end_time = duration
            
            # Trim video
            output_path = os.path.join(temp_dir, f"trimmed_{int(time.time())}.mp4")
            temp_files.append(output_path)
            
            log_progress(f"✂️ Trimming from {start_time}s to {end_time}s...", True)
            trim_video_fast(main_filename, output_path, start_time, end_time)
            
            final_duration = get_video_duration(output_path)
            log_progress(f"✅ Trim complete! Final duration: {final_duration:.1f} seconds", True)
            
        elif operation == 'merge':
            log_progress("🔗 MERGE WITH INTRO", True)
            
            # Predefined intro video
            intro_id = intro_video_id or "1-BRZ8NCrVkw1HuHkLwz5PRC8zIxsb99n"
            main_file_id = extract_gdrive_file_id(video1_url)
            
            # Download intro
            intro_path = os.path.join(temp_dir, f"intro_{int(time.time())}.mp4")
            temp_files.append(intro_path)
            
            if not download_gdrive_video(intro_id, intro_path, "intro"):
                return
            
            # Download main
            main_path = os.path.join(temp_dir, f"main_{int(time.time())}.mp4")
            temp_files.append(main_path)
            
            if not download_gdrive_video(main_file_id, main_path, "main video"):
                return
            
            # Merge with intro
            output_path = os.path.join(temp_dir, f"merged_{int(time.time())}.mp4")
            temp_files.append(output_path)
            
            success = merge_with_intro_only(intro_path, main_path, output_path)
            
            if not success:
                return
            
            final_duration = get_video_duration(output_path)
            log_progress(f"✅ Merge complete! Final duration: {final_duration:.1f} seconds", True)
            
        elif operation == 'trim_and_merge':
            log_progress("✂️🔗 TRIM AND MERGE WITH INTRO", True)
            
            # Predefined intro video
            intro_id = intro_video_id or "1-BRZ8NCrVkw1HuHkLwz5PRC8zIxsb99n"
            main_file_id = extract_gdrive_file_id(video1_url)
            
            # Download intro
            intro_path = os.path.join(temp_dir, f"intro_{int(time.time())}.mp4")
            temp_files.append(intro_path)
            
            if not download_gdrive_video(intro_id, intro_path, "intro"):
                return
            
            # Download main
            main_path = os.path.join(temp_dir, f"main_{int(time.time())}.mp4")
            temp_files.append(main_path)
            
            if not download_gdrive_video(main_file_id, main_path, "main video"):
                return
            
            # Get video info
            intro_duration = get_video_duration(intro_path)
            main_duration = get_video_duration(main_path)
            
            log_progress(f"📊 Video Info:", True)
            log_progress(f"Intro: {intro_duration:.1f} seconds", True)
            log_progress(f"Main: {main_duration:.1f} seconds", True)
            
            # Validate times
            if end_time is None:
                end_time = main_duration
            
            if start_time < 0:
                start_time = 0
            if end_time > main_duration:
                end_time = main_duration
            if end_time <= start_time:
                end_time = start_time + 10
            
            # Calculate expected duration
            expected_duration = (end_time - start_time) + intro_duration
            log_progress(f"📋 Expected output: {expected_duration:.1f} seconds", True)
            
            # Create final video
            output_path = os.path.join(temp_dir, f"trimmed_merged_{int(time.time())}.mp4")
            temp_files.append(output_path)
            
            log_progress(f"🎬 Processing...", True)
            success = create_final_video_fast(intro_path, main_path, start_time, end_time, output_path)
            
            if not success:
                return
            
            final_duration = get_video_duration(output_path)
            log_progress(f"✅ Trim & Merge complete! Final duration: {final_duration:.1f} seconds", True)
            
            # Check if duration matches expected
            if abs(final_duration - expected_duration) > 1:
                log_progress(f"⚠️ Duration mismatch! Expected: {expected_duration:.1f}s, Got: {final_duration:.1f}s", True)
            else:
                log_progress(f"✅ Duration matches expected!", True)
            
            # Check audio
            if has_audio_stream(output_path):
                log_progress(f"🔊 Audio: PRESENT ✓", True)
            else:
                log_progress(f"🔊 Audio: MISSING ✗", True)
        
        elif operation == 'merge_custom':
            log_progress("🔗 MERGE TWO CUSTOM VIDEOS", True)
            
            # Extract and download first video
            video1_id = extract_gdrive_file_id(video1_url)
            video1_path = os.path.join(temp_dir, f"video1_{int(time.time())}.mp4")
            temp_files.append(video1_path)
            
            if not download_gdrive_video(video1_id, video1_path, "first video"):
                return
            
            # Extract and download second video
            video2_id = extract_gdrive_file_id(video2_url)
            video2_path = os.path.join(temp_dir, f"video2_{int(time.time())}.mp4")
            temp_files.append(video2_path)
            
            if not download_gdrive_video(video2_id, video2_path, "second video"):
                return
            
            # Merge videos
            output_path = os.path.join(temp_dir, f"merged_{int(time.time())}.mp4")
            temp_files.append(output_path)
            
            success = merge_two_custom_videos(video1_path, video2_path, output_path)
            
            if not success:
                return
            
            final_duration = get_video_duration(output_path)
            log_progress(f"✅ Merge complete! Final duration: {final_duration:.1f} seconds", True)
        
        else:
            log_progress(f"❌ Invalid operation: {operation}", True)
            return
        
        # ===============================================
        # UPLOAD TO GOOGLE DRIVE
        # ===============================================
        if output_path and os.path.exists(output_path):
            file_size = os.path.getsize(output_path) / (1024 * 1024)
            final_duration = get_video_duration(output_path)
            
            log_progress(f"📦 File ready: {file_size:.2f} MB, {final_duration:.1f} seconds", True)
            log_progress("☁️ Uploading to Google Drive...", True)
            
            drive_url = upload_to_drive(output_path, chat_id)
            
            if drive_url:
                final_msg = f"""
<b>✅ Video Editing Complete!</b>

🎬 Operation: {operation}
📏 Size: {file_size:.2f} MB
⏱️ Duration: {final_duration:.1f} seconds

🔗 <b>Download Link:</b>
{drive_url}
                """
                send_telegram_message(chat_id, final_msg)
                log_progress(f"✅ Upload successful!", False)
            else:
                log_progress("❌ Upload failed", True)
        
    except Exception as e:
        logger.error(f"[VIDEO EDITING] Error: {e}")
        error_msg = f"""
<b>❌ Video Editing Failed</b>

Error: {str(e)[:200]}

Possible issues:
• Invalid Google Drive link
• FFmpeg processing error
• Time values out of range
"""
        send_telegram_message(chat_id, error_msg)
    
    finally:
        # Cleanup temp files
        for file_path in temp_files:
            if file_path and os.path.exists(file_path):
                try:
                    os.remove(file_path)
                except:
                    pass
        
        # Clean up any concat files
        temp_concat_files = [f for f in os.listdir(temp_dir) if f.startswith('concat_')]
        for concat_file in temp_concat_files:
            try:
                os.remove(os.path.join(temp_dir, concat_file))
            except:
                pass
